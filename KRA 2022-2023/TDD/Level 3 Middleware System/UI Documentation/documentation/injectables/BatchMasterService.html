  getAllCreatedBatch(paginationParams:PaginationParams) {

    let params : any = new HttpParams();

    if(paginationParams.pageIndex){
      params = params.append('pageIndex',paginationParams.pageIndex);
    }

    if(paginationParams.pageSize){
      params = params.append('pageSize',paginationParams.pageSize);
    }

    if(paginationParams.search)
    {
      params = params.append('search',paginationParams.search);
    }
    if(paginationParams.fromDate)
    {
      params = params.append('fromDate',paginationParams.fromDate);
    }
    if(paginationParams.toDate)
    {
      params = params.append('toDate',paginationParams.toDate);
    }
    return this.http.get<IResult>(this.apiUrl + '/batch-master-list',{params:params}).pipe(
      map((x: IResult) => x),
      catchError((error: Response) => {
        return throwError(error);
      })
    );
  }

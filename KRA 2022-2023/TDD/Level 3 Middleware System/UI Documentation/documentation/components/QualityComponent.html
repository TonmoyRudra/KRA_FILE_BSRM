import { GlobalService } from 'src/app/services/global/global.service';
import { QualityResult } from './../../../../shared/models/QualityResultDto';
import { QualityResultService } from './../../../../services/qualityResult/quality-result.service';
import { Component, Input, OnInit } from '@angular/core';
import { BatchDetailsDto } from 'src/app/shared/models/BatchDetailsDto';
import { BatchMaster } from 'src/app/shared/models/BatchMaster';
import { PaginationParams } from 'src/app/shared/models/paginationParams';
import { BATCH_STEP_NO } from 'src/app/shared/enums/batch.enum';

@Component({
  selector: 'app-quality',
  templateUrl: './quality.component.html',
  styleUrls: ['./quality.component.css']
})
export class QualityComponent implements OnInit {
  @Input() batchDetailsList: BatchDetailsDto[];
  @Input() batchMasterInfo: BatchMaster;
  @Input() isEditMode: boolean;
  paginationParams: PaginationParams = new PaginationParams();

  scrollByContent = true;
  scrollByThumb = true;
  scrollbarMode: string = 'always'; //  'onScroll' | 'onHover' | 'always' | 'never'

  selecteditemCode:string;
  qualityResultList: QualityResult[];
  distrintQualityTestColumn: string[];
  qualityArrayList: any[];
  columns: any;
  constructor(public qualityResultService: QualityResultService,
    public globalService : GlobalService) { }

  ngOnInit(): void {
    this.refreshDataGrid();
  }

  onToolbarPreparing(e) {
    e.toolbarOptions.items.unshift({
      location: 'after',
      widget: 'dxButton',
      options: {
        icon: 'refresh',
        onClick: this.refreshDataGrid.bind(this),
      },
    });
  }

  refreshDataGrid() {
    this.getQualityResultById();
    //this.dataGrid.instance.refresh();
  }

  getQualityResultById() {
    this.paginationParams.midFk = this.batchMasterInfo.midPk;
    this.qualityResultService.getQualityDataById(this.paginationParams).subscribe((result) => {
      if (result.statusCode == 200) {
        this.qualityResultList = result.values;
        // Filter Batch Step 10,20,30 only.
        this.qualityResultList = this.qualityResultList.filter(x => x.batchStepNo == BATCH_STEP_NO.IF || x.batchStepNo == BATCH_STEP_NO.LRF || x.batchStepNo == BATCH_STEP_NO.CCM)
        console.log(this.qualityResultList);
        this.makeQualityList();
      }
    }, err => {
      this.globalService.errorResponseHandler(err);
    })
  }

  makeQualityList(){
    // this.distrintQualityTestColumn.push('midFk');
    this.distrintQualityTestColumn = [...new Set(this.qualityResultList.map((x) => x.testName))]; // Distinct
    this.distrintQualityTestColumn.unshift('source');
    console.log(this.distrintQualityTestColumn);


    // Make Object from Array
    // For 10
    let testObj: any =  this.distrintQualityTestColumn.reduce((acc,curr)=> (acc[curr]='',acc),{});

    this.qualityArrayList = [];
    this.qualityResultList.filter(x => x.batchStepNo == 10).forEach(element => {
      // testObj['midFk'] = element.midFk;
      testObj['source'] = 'IF';
      testObj[element.testName] =  element.resultValue;
    });
    this.qualityArrayList.push(testObj);


    let testObj2: any =  this.distrintQualityTestColumn.reduce((acc,curr)=> (acc[curr]='',acc),{});

    this.qualityResultList.filter(x => x.batchStepNo == 20).forEach(element => {
      //testObj2['midFk'] = element.midFk;
      testObj2['source'] =  "LRF"
      testObj2[element.testName] =  element.resultValue;
    });
    this.qualityArrayList.push(testObj2);


    let testObj3: any =  this.distrintQualityTestColumn.reduce((acc,curr)=> (acc[curr]='',acc),{});

    this.qualityResultList.filter(x => x.batchStepNo == 30).forEach(element => {
      //testObj3['midFk'] = element.midFk;
      testObj3['source'] = "CCM"
      testObj3[element.testName] =  element.resultValue;
    });
    this.qualityArrayList.push(testObj3);


    console.log(this.qualityArrayList);



  }
}

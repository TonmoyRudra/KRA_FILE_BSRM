import { GlobalParam } from './../../../../shared/models/GlobalParam';
import { Component, OnInit } from '@angular/core';
import CustomStore from 'devextreme/data/custom_store';
import { AuthService } from 'src/app/services/auth/auth.service';
import { BatchMasterService } from 'src/app/services/batchMaster/batch-master.service';
import { GlobalService } from 'src/app/services/global/global.service';
import { PaginationParams } from 'src/app/shared/models/paginationParams';

@Component({
  selector: 'app-global-param',
  templateUrl: './global-param.component.html',
  styleUrls: ['./global-param.component.css']
})
export class GlobalParamComponent implements OnInit {

  sessionUser: any;
  dataSource: any = [];
  paginationParams: PaginationParams = new PaginationParams();
  sortValues = [
    { name: '10', value: '10' },
    { name: '20', value: '20' },
    { name: '30', value: '30' },
    { name: '40', value: '40' },
  ];
  result: any;
  resultList: GlobalParam [] = [];

  constructor(
    private authService: AuthService,
    private globalService: GlobalService,
    public batchMasterService: BatchMasterService) {
    this.sessionUser = this.authService.getSessionUser();
  }

  ngOnInit(): void {
    // this.paginationParams.pageSize = 2;
    this.getGlobalParamList();
  }
  onToolbarPreparing(e) {
    e.toolbarOptions.items.unshift({
      location: 'after',
      widget: 'dxButton',
      options: {
        icon: 'refresh',
        onClick: this.refreshDataGrid.bind(this),
      },
    });
  }
  refreshDataGrid() {
    this.getGlobalParamList();
    //this.dataGrid.instance.refresh();
  }

  async getGlobalParamList() {
    (await this.batchMasterService.getGlobalParamList(this.paginationParams)).subscribe(
      (result) => {
        if (result.statusCode == 200) {
          // this.programList = result.resultset;
          this.result = result;
          this.resultList = result.values;
          let slNo = 1;
          result.values.forEach((element: any) => {
            element.slNo = slNo++;
          });

          this.generateDataGrid(result.values);
        }
      },
      (err) => {
        console.log(err);
      }
    );
  }
  onPageSizeSelectedChanged(event: any) {
    this.paginationParams.pageSize = event.target.value;
    this.getGlobalParamList();
  }

  addPageEvent(event: any) {
    if (this.paginationParams.pageIndex !== event) {
      this.paginationParams.pageIndex = event;
      this.getGlobalParamList();
    }
  }

  generateDataGrid(list) {
    this.dataSource = new CustomStore({
      key: 'gvidPk',
      load: () => list,
      insert: (values) => this.create(values),
      update: (key, updateBody) => this.update(key, updateBody),
      remove: (key) => this.delete(key)
    });
  }

  update(key, updateBody: any): any {

    console.log('Update', key, updateBody);
    var selectedData = this.resultList.filter((s) => s.gvidPk === key)[0];
    // this.dataSource.byKey(key).then((dataItem) => {
    //   selectedData = dataItem;

    // }, (error) => {
    // selectedData = [];

    // })

    for (const key in updateBody) {
      if (selectedData.hasOwnProperty(key)) {
        selectedData[key] = updateBody[key];
      }
    }
    console.log('Update', selectedData);
    this.globalService.showSpinner(true);
    this.batchMasterService.updateGlobalParam(selectedData).subscribe(result => {
      if (result.statusCode == 200) {
        this.globalService.showSpinner(false);
        this.globalService.showSwal('Updated' , 'Succesfully Updated.', 'success');
        this.getGlobalParamList();
      }

    }, err => {
      console.log(err);
      this.globalService.showSpinner(false);
      this.globalService.showSwal('Warning', err.Message, 'error');
    })
  }

import { Component, EventEmitter, Input, OnInit, Output } from '@angular/core';
import { AuthService } from 'src/app/services/auth/auth.service';
import { BatchMasterService } from 'src/app/services/batchMaster/batch-master.service';
import { GlobalService } from 'src/app/services/global/global.service';
import { ItemMapService } from 'src/app/services/item-map/item-map.service';
import { PhysicalInspectionService } from 'src/app/services/physical-inspection/physical-inspection.service';
import { BatchDetailsDto } from 'src/app/shared/models/BatchDetailsDto';
import { BatchMaster } from 'src/app/shared/models/BatchMaster';
import { PaginationParams } from 'src/app/shared/models/paginationParams';

@Component({
  selector: 'app-validation-failed-popup',
  templateUrl: './validation-failed-popup.component.html',
  styleUrls: ['./validation-failed-popup.component.css']
})
export class ValidationFailedPopupComponent implements OnInit {
  @Input() batchMasterInfo: BatchMaster;
  @Output() closePopupClick = new EventEmitter<any>();

  sessionUser: any;
  paginationParams: PaginationParams = new PaginationParams();

  scrollByContent = true;
  scrollByThumb = true;
  scrollbarMode: string = 'always'; //  'onScroll' | 'onHover' | 'always' | 'never'
  batchDetailList: BatchDetailsDto[];
  isEditMode : boolean = false;

  constructor(
    private authService: AuthService,
    private globalService: GlobalService,
    public itemMapService:ItemMapService,
    public batchMasterService: BatchMasterService,
    public physicalInspectionService: PhysicalInspectionService) {
    this.sessionUser = this.authService.getSessionUser();

  }

  ngOnInit(): void {
    this.refreshDataGrid();
  }
  onToolbarPreparing(e) {
    e.toolbarOptions.items.unshift({
      location: 'after',
      widget: 'dxButton',
      options: {
        icon: 'refresh',
        onClick: this.refreshDataGrid.bind(this),
      },
    },
    );
  }
  refreshDataGrid() {
    this.getIngredientsfromBatch();
    //this.dataGrid.instance.refresh();
  }

  getIngredientsfromBatch() {
    this.paginationParams.midFk = this.batchMasterInfo.midPk;
    this.batchMasterService.getBatchDetails(this.paginationParams).subscribe(result => {
      if (result.statusCode == 200) {
        this.batchDetailList = result.values;
        this.batchDetailList = this.batchDetailList.filter(x => x.errorMsg != null);
        let slNo = 1;
        this.batchDetailList.forEach((element) => {
          element.slNo = slNo++;
        });
        console.log(this.batchDetailList);
      }
    }, err => {
      this.globalService.errorResponseHandler(err);
    })
  }
}

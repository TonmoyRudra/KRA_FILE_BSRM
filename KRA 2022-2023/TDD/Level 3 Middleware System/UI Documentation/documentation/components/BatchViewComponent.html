import { BATCH_STATUS } from 'src/app/shared/enums/batch.enum';
import { PhysicalInspectionService } from './../../../../services/physical-inspection/physical-inspection.service';
import { GlobalParam } from 'src/app/shared/models/GlobalParam';
import { PhysicalInspection } from './../../../../shared/models/PhysicalInspection';
import { ItemMapService } from './../../../../services/item-map/item-map.service';
import { PaginationParams } from './../../../../shared/models/paginationParams';
import { BatchMaster } from './../../../../shared/models/BatchMaster';
import { BatchMasterService } from './../../../../services/batchMaster/batch-master.service';
import { AfterViewInit, Component, ComponentRef, ElementRef, EventEmitter, Input, OnDestroy, OnInit, Output } from '@angular/core';
import { AuthService } from 'src/app/services/auth/auth.service';
import { GlobalService } from 'src/app/services/global/global.service';
import { BatchDetailsDto } from 'src/app/shared/models/BatchDetailsDTO';
@Component({
  selector: 'app-batch-view',
  templateUrl: './batch-view.component.html',
  styleUrls: ['./batch-view.component.css']
})
export class BatchViewComponent implements OnInit {
  @Input() batchMasterInfo: BatchMaster;
  @Output() closePopupClick = new EventEmitter<any>();

  sessionUser: any;
  paginationParams: PaginationParams = new PaginationParams();

  physicalInspectionModel : PhysicalInspection [] = [];
  billetTypeModel : PhysicalInspection [] = [];
  unitValue: GlobalParam;
  mergeModel: any [] = [];


  // DX Scroll View Config
  scrollByContent = true;
  scrollByThumb = true;
  scrollbarMode: string = 'always'; //  'onScroll' | 'onHover' | 'always' | 'never'
  batchDetailList: BatchDetailsDto[];
  isEditMode : boolean = false;

  constructor(
    private authService: AuthService,
    private globalService: GlobalService,
    public itemMapService:ItemMapService,
    public batchMasterService: BatchMasterService,
    public physicalInspectionService: PhysicalInspectionService) {
    this.sessionUser = this.authService.getSessionUser();

  }

  ngOnInit(): void {
    if(this.batchMasterInfo.status == BATCH_STATUS.VALIDATION_FAILED){
      this.isEditMode = true;
    } else {
      this.isEditMode = false;
    }
    if(this.batchMasterInfo != null){
      this.getBatchDetails();
    }
  }

  getBatchDetails(){

    this.paginationParams.orgId = this.batchMasterInfo.orgId;
    this.paginationParams.midFk = this.batchMasterInfo.midPk;


    this.batchMasterService.getBatchDetails(this.paginationParams).subscribe(result => {
      if (result.statusCode == 200) {
        //this.result = result;
        this.batchDetailList = result.values; 
      }
    }, err => {
      this.globalService.errorResponseHandler(err);
      console.log(err);
    })
  }


  processValidation(e){
    this.paginationParams.midFk = this.batchMasterInfo.midPk;
    this.physicalInspectionService.processValidation(this.paginationParams).subscribe(result =>{
      if (result.statusCode == 200) {
        //this.result = result;
        console.log(result);
        this.globalService.showSwal('Success','Re-Process Successfully Done','success');
        this.batchMasterService.loadBatchListObs(true);
      }
    }, err => {
      this.globalService.errorResponseHandler(err);
      console.log(err);
    })
  }
}

  async getGlobalParamList() {
    this.paginationParams.pageSize = 200;
    (await this.batchMasterService.getGlobalParamList(this.paginationParams)).subscribe(
      (result) => {
        if (result.statusCode == 200) {
          // this.result = result;
          this.batchMasterService.globalParamData = result.values;

  generatedPhyInsList_SaveMode(){
    this.physicalInspectionModel = [];
    this.billetTypeModel = [];

    let obj : PhysicalInspection ;
    this.lengthList.forEach(element => {
      obj = new PhysicalInspection();

      obj.lengthInMm = element.lengthInMm;
      obj.pInsType = PHYSICAL_INSPECTION_TYPE.CONFORMING_NONCONFIRMING;
      obj.isPrime = false;
      if(obj.lengthInMm == 12000){
        // if 12 m Billet. it should be auto Prime 
        obj.isPrimeProduct = 1;
        obj.isPrime = true;
      }
      this.physicalInspectionModel.push(obj);
    });

    // Billat Type Add
    // Short Billet
      var shortBilletObj : PhysicalInspection = new PhysicalInspection();
      shortBilletObj.lengthInMm = 0;
      shortBilletObj.pInsType = PHYSICAL_INSPECTION_TYPE.SHORT_BILLET;
      this.billetTypeModel.push(shortBilletObj);

      // Crop Billet
      var cropBilletObj : PhysicalInspection = new PhysicalInspection();
      cropBilletObj.lengthInMm = 0;
      cropBilletObj.pInsType = PHYSICAL_INSPECTION_TYPE.CROP_BILLET;
      this.billetTypeModel.push(cropBilletObj);

      console.log(this.physicalInspectionModel)
      console.log(this.billetTypeModel);
  }

  generatedPhyInsList_ViewMode(data : PhysicalInspection[]){
    this.physicalInspectionModel = [];
    this.billetTypeModel = [];
    data.forEach(element => {
      if(element.pInsType == PHYSICAL_INSPECTION_TYPE.CONFORMING_NONCONFIRMING){
        element.lengthInMm = element.lengthInMm
        element.isPrime = element.isPrimeProduct == 1 ? true : false;
        this.physicalInspectionModel.push(element);
      } else {
        //CROP_BILLET & SHORT_BILLET
        this.billetTypeModel.push(element);
      }
    });
  }

  onCellPreparingBilletGrid(e) {
    if(!this.isViewMode){
      if(e.rowType === "data" && (e.column.dataField === "totalPcs")) {
        e.isEditing = e.row.data.pInsType == 'SHORT_BILLET' ? false : true;
        if(e.isEditing == false) {
          e.cellElement.style.cssText = 'background: #00968840;padding: 2px;text-align: center;border: 1px solid;border-color: #009688;'; 
        }
      }
      if(e.rowType === "data" && (e.column.dataField === "lengthInMm")) { 
        e.cellElement.style.cssText = 'background: #00968840;padding: 2px;text-align: center;border: 1px solid;border-color: #009688;'; 
      }
    }
  }

  phyInspCellDataChanged(e, cell) {
    //dx-datagrid-headers
    //cell.setValue(e.value, "forecastQ1");
    this.unitValue = this.batchMasterService.getGlobalParamValue(GLOBAL_PARAM.PER_METER_WEIGHT_KG);

    // (Total lengthInMM * PCs)/1000 * PER_METER_WEIGHT_KG
    cell.data.conKgs = ((cell.data.lengthInMm * cell.data.conPcs) / 1000) * this.unitValue.paramValueNumber;
    cell.data.nonconKgs = ((cell.data.lengthInMm * cell.data.nonconPcs) / 1000) *  this.unitValue.paramValueNumber;

    cell.data.totalPcs = cell.data.nonconPcs + cell.data.conPcs;
    cell.data.totalKgs =  cell.data.nonconKgs +   cell.data.conKgs ;


    //this.generateCompanyWiseTotal();
    //console.log(this.dataSource);
  }

  onSubmit(e){
    this.globalService.showSpinner(true);
    console.log(this.physicalInspectionModel)
    console.log(this.billetTypeModel);
    let isValidate: boolean = false; // validate for any totalKgs found or not.
    let isPrimeValidate: boolean = false;
    this.mergeModel = [];
    this.mergeModel.push(...this.physicalInspectionModel)
    this.mergeModel.push(...this.billetTypeModel);

    this.mergeModel.forEach


    let countConKgs = 0;
    let countConKgsIndex;
    this.mergeModel.forEach((element, index) => {
      if(element.conKgs > 0){
        countConKgs += 1;
        countConKgsIndex = index;
      }
    });
    if(countConKgs == 1){
      // Only one data inputed. and make it by default Prime product.
      this.mergeModel[countConKgsIndex].isPrimeProduct = 1;
    }

    let primeCount = 0;
    this.mergeModel.forEach((element: PhysicalInspection) => {
      element.batchNumber = this.batchMasterInfo.batchNumber;
      element.midFk = this.batchMasterInfo.midPk;
      element.orgId = this.batchMasterInfo.orgId;
      element.grade = this.grade;
      element.dia = this.dia;
      element.perMtrWeightKgs = this.unitValue.paramValueNumber;

      if(element.totalKgs > 0){
         isValidate = true;
      }

      if(element.isPrimeProduct == 1){
        isPrimeValidate = true;
        primeCount = primeCount + 1;
      }

      //if(element.lengthInMm == 12000){
      //   // if 12 m Billet. it should be auto Prime
      //   isPrimeValidate = true;

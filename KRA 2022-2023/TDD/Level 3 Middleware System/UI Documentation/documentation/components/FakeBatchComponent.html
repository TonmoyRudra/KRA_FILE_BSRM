import { Component, OnInit } from '@angular/core';
import { AuthService } from 'src/app/services/auth/auth.service';
import { BatchMasterService } from 'src/app/services/batchMaster/batch-master.service';
import { GlobalService } from 'src/app/services/global/global.service';
import { BATCH_STATUS } from 'src/app/shared/enums/batch.enum';
import { BatchMaster } from 'src/app/shared/models/BatchMaster';
import { PaginationParams } from 'src/app/shared/models/paginationParams';
import { IResult } from 'src/app/shared/models/result';
import Swal from 'sweetalert2';

@Component({
  selector: 'app-fake-batch',
  templateUrl: './fake-batch.component.html',
  styleUrls: ['./fake-batch.component.css']
})
export class FakeBatchComponent implements OnInit {

  batchNumber: any;
  sessionUser: any;
  batch_status_enum: typeof BATCH_STATUS;
  paginationParams: PaginationParams = new PaginationParams();
  result: IResult;
  dataSource: any = [];
  batchMasterInfo: BatchMaster;

  constructor(
    private authService: AuthService,
    private globalService: GlobalService,
    public batchMasterService:BatchMasterService) {
    this.sessionUser = this.authService.getSessionUser();

    this.batch_status_enum = BATCH_STATUS;


  }

  ngOnInit(): void {
  }

  onSubmit(e){
    console.log(e);
    //this.getBatchMasterByNumber();
    if(this.batchMasterInfo.status == BATCH_STATUS.INCOMPLETE){
      Swal.fire({
        title: 'Are you sure?',
        text: 'You want to make this batch as FAKE or WRONG batch?',
        icon: 'warning',
        showCancelButton: true,
        confirmButtonColor: '#3085d6',
        cancelButtonColor: '#d33',
        confirmButtonText: 'Yes',
      }).then((result) => {
        /* Read more about isConfirmed, isDenied below */
        if (result.isConfirmed) {
          this.globalService.showSpinner(true);
          this.batchMasterInfo.status = BATCH_STATUS.FAKE_BATCH;
          this.updateBatchMaster(this.batchMasterInfo);
        }
      });
    } else {
      this.globalService.showSwal('Batch Status:'+this.batchMasterInfo.status,'You cannot make this batch a FAKE Batch. You can only change it as a fake batch when it was in the INCOMPLETE stage.', 'warning', false);
    }
  }

  updateBatchMaster(updateModel: BatchMaster){
    this.batchMasterService.updateBatchMasterData(updateModel).subscribe(result => {
      if (result.statusCode == 200) {
        this.globalService.showSpinner(false);

        console.log(result);
        this.globalService.showSwal('Success', 'This batch status is updated by FAKE BATCH', 'success')
        this.batchMasterInfo = null;
        this.dataSource = [];
        this.batchNumber = null;
        this.batchMasterService.loadBatchListObs(true);
        // this.closePopupClick.emit();
        // this.globalService.showSwal('Success', 'Physical Inspection Done Successfully.', 'success');
        // let slNo = 1;
        // this.gradeList.forEach((element) => {
        //   element.slNo = slNo++;
        // });
      }
    }, error => {
      this.globalService.showSpinner(false);
      this.globalService.errorResponseHandler(error.error)
    })
  }

  async getBatchMasterByNumber() {
    this.dataSource = [];
    this.paginationParams.batchNumber = this.batchNumber;
    (await this.batchMasterService.getBatchMasterByNumber(this.paginationParams)).subscribe(
      (result : IResult) => {
        if (result.statusCode == 200) {
          // this.programList = result.resultset;
          this.result = result;
          this.dataSource = result.values;
          if(this.dataSource.length != 0){
            console.log(this.dataSource);
            this.batchMasterInfo = this.dataSource[0];
          } else {
            //this.globalService.showSwal('Not Found','No batch found with this batch number.', 'warning');

          }
        }
      },
      (err) => {
        this.globalService.errorResponseHandler(err)
      }
    );
  }

  showWarning(){
    this.globalService.showSwal('Warning','This batch is not elegible for to make as a fake or wrong batch.', 'warning');
  }

  pinEnterKeyPress($event) {
    let value = (<HTMLInputElement>event.target).value;
    if (<HTMLInputElement>event.target) {
      // if (value.length > 4) {
      //   value = value.slice(0, 4)
      // }
      //  (<HTMLInputElement>event.target).value = value.replace(/\D/g, ''); // remove all space and char
      (<HTMLInputElement>event.target).value = value.replace(/\s/g, ''); // remove all space
      this.batchNumber = (<HTMLInputElement>event.target).value;
      if(this.batchNumber){
        this.getBatchMasterByNumber()
      }
    }

    // if ($event.event.key === 'Enter') {
    //   this.getBatchMasterByNumber()
    // }
  }
}

import { ProcessParameterService } from './../../../../services/process-parameter/process-parameter.service';
import { Resource } from './../../../../shared/models/ResourceDto';
import { BATCH_STEP_NO } from './../../../../shared/enums/batch.enum';
import { ResourceService } from './../../../../services/Resource/resource.service';
import { Component, Input, OnInit } from '@angular/core';
import { BatchMasterService } from 'src/app/services/batchMaster/batch-master.service';
import { GlobalService } from 'src/app/services/global/global.service';
import { BatchDetailsDto } from 'src/app/shared/models/BatchDetailsDto';
import { BatchMaster } from 'src/app/shared/models/BatchMaster';
import { PaginationParams } from 'src/app/shared/models/paginationParams';
import CustomStore from 'devextreme/data/custom_store';
import { ProcessParameter } from 'src/app/shared/models/ProcessParameterDto';

@Component({
  selector: 'app-process-parameter',
  templateUrl: './process-parameter.component.html',
  styleUrls: ['./process-parameter.component.css']
})
export class ProcessParameterComponent implements OnInit {

  @Input() batchDetailsList: BatchDetailsDto[];
  @Input() isEditMode: boolean;
  @Input() batchMasterInfo: BatchMaster;
  paginationParams: PaginationParams = new PaginationParams();
  ingredientListFromBatchDetails: BatchDetailsDto[];

  // DX Scroll View Config
  scrollByContent = true;
  scrollByThumb = true;
  scrollbarMode: string = 'always'; //  'onScroll' | 'onHover' | 'always' | 'never'

  processParamList: any;
  dataSource: CustomStore;

  constructor(public batchMasterService: BatchMasterService,
    public globalService: GlobalService,
    public processService: ProcessParameterService) {
  }

  ngOnInit(): void {

    this.refreshDataGrid();
  }

  onToolbarPreparing(e) {
    e.toolbarOptions.items.unshift({
      location: 'after',
      widget: 'dxButton',
      options: {
        icon: 'refresh',
        onClick: this.refreshDataGrid.bind(this),
      },
    }
    );
  }
  refreshDataGrid() {
    this.getProcessParameterById();
    // this.getAllItemLineTypeWise(this.batchMasterInfo.orgId, MATERIAL_TYPE.INGREDIENT);
    this.datagrid_DXSaveButton_Disable();
    //this.dataGrid.instance.refresh();
  }

  datagrid_DXSaveButton_Disable(){
    let dxSaveButton: any = document.getElementsByClassName('dx-datagrid-save-button')[0];

    if(dxSaveButton){
      dxSaveButton.style.display = 'none';
    }
  }

  datagrid_DXSaveButton_EnableClick(){
    let dxSaveButton: any ;
    dxSaveButton = document.getElementsByClassName('dx-datagrid-save-button')[0];
    if(dxSaveButton){
      dxSaveButton.style.display = 'none';
      dxSaveButton.click();
    }
  }

  getProcessParameterById() {
    this.paginationParams.midFk = this.batchMasterInfo.midPk;
    this.processService.getDataById(this.paginationParams).subscribe(result => {
      if (result.statusCode == 200) {
        this.processParamList = result.values;
        let slNo = 1;
        this.processParamList.forEach((element) => {
          element.slNo = slNo++;
        });
        console.log(this.processParamList);
        this.generateDataGrid(this.processParamList);
      }
    }, err => {
      this.globalService.errorResponseHandler(err);
    })
  }

  getBatchStepName(data){
    if(data.batchStepNo == BATCH_STEP_NO.IF) return 'IF';
    if(data.batchStepNo == BATCH_STEP_NO.LRF) return 'LRF';
    if(data.batchStepNo == BATCH_STEP_NO.CCM) return 'CCM';
    if(data.batchStepNo == BATCH_STEP_NO.PHYSICAL_INSPECTION) return 'PHYSICAL INSPECTION';
  }

  // onSubmit(e) {
  //   this.datagrid_DXSaveButton_EnableClick();
  //   console.log(e);
  //   console.log(this.processParamList);

  //   this.processService.upSertResouce(this.processParamList).subscribe(result => {
  //     if (result.statusCode == 200) {
  //      // this.resourceList = result.values;
  //       this.globalService.showSwal('Success', 'Successfully saved.', 'success');
  //     }
  //   }, err => {
  //     this.globalService.errorResponseHandler(err);
  //   })
  // }

  update(key, updateBody) : any{
    //this.datagrid_DXSaveButton_EnableClick();
    console.log(this.processParamList);

    var selectedData: ProcessParameter = this.processParamList.filter((s) => s.rsintidPk === key)[0];

    for (const key in updateBody) {
      if (selectedData.hasOwnProperty(key)) {
        selectedData[key] = updateBody[key];
      }
    }

    this.processService.upSertResouce([selectedData]).subscribe(result => {
      if (result.statusCode == 200) {
       // this.resourceList = result.values;
        this.globalService.showSwal('Success', 'Successfully saved.', 'success');
        this.refreshDataGrid();
      }
    }, err => {
      this.globalService.errorResponseHandler(err);
    })
  }

  generateDataGrid(list) {
    // var sl = 1;
    // list.forEach(element => {
    //   element.sl = sl++;
    // });

    this.dataSource = new CustomStore({
      key: 'rsintidPk',
      load: () => list,
      //insert: (values) => this.storeSaveDatasoruce(values),
      update: (key, updateBody) => this.update(key, updateBody),
      //remove: (key) => this.deleteOrg(key)
    });
  }
}

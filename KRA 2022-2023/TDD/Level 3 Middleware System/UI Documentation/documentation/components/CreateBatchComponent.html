      if(result == true){
        this.getCreatedBatchList();
      }
    })
  }

  ngOnDestroy() {
    this.unsubscribe$.next(true);
    this.unsubscribe$.complete();

async getCreatedBatchList() {
  this.globalService.showSpinner(true);
  if(this.fromDate){
    var fromDate = `${this.globalService.formattedDate(this.fromDate.toDateString())}`;//.split("T")[0];
    this.paginationParams.fromDate = fromDate;
  }
  if(this.toDate){
    var toDate = `${this.globalService.formattedDate(this.toDate.toDateString())}`;//.split("T")[0];
    this.paginationParams.toDate = toDate;
  }

  (await this.batchMasterService.getAllCreatedBatch(this.paginationParams)).subscribe(
    (result : IResult) => {
      if (result.statusCode == 200) {
      this.globalService.showSpinner(false);

        // this.programList = result.resultset;
        this.result = result;
        this.dataSource = result.values;
        if(!this.isFakeBatch) this.dataSource = this.dataSource.filter(x => x.status != BATCH_STATUS.FAKE_BATCH);
        if(this.isFakeBatch)  this.dataSource = this.dataSource.filter(x => x.status == BATCH_STATUS.FAKE_BATCH);
        this.dataSource = this.dataSource.sort(this.globalService.dynamicSort('ifStartDate'));

        let slNo = 1;
        this.dataSource.forEach((element) => {
          element.slNo = slNo++;
        });
      }
    },
    (err) => {
  this.globalService.showSpinner(false);

      console.log(err);

    }
  );
}

onPageSizeSelectedChanged(event: any) {
  this.paginationParams.pageSize = event.target.value;
  this.getCreatedBatchList();
}

addPageEvent(event: any) {
  if (this.paginationParams.pageIndex !== event) {
    this.paginationParams.pageIndex = event;
    this.getCreatedBatchList();
  }
}

getStatusColorClassName(stat: string){
  if(stat == this.batch_status_enum.PHYSICAL_INSPECTION_PENDING){
    return 'phyInsPending_BGColor';
  }
  else if(stat == this.batch_status_enum.ERROR || stat == this.batch_status_enum.FAILED){
    return 'error_BGColor';
  }
  else if(stat == this.batch_status_enum.VALIDATION_FAILED){
    return 'validationFailed_BGColor';
  }
  else if(stat == this.batch_status_enum.INCOMPLETE  ){
    return 'incomplete_BGColor';
  }
  else if(stat == this.batch_status_enum.WAITING){
    return 'waiting_BGColor';
  }
  else if(stat == this.batch_status_enum.CLOSE){
    return 'close_BGColor';
  }else {
    return 'common_BGColor'
  }
}

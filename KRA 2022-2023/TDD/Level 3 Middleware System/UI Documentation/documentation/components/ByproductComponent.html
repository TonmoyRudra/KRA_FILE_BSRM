import { ItemMapService } from './../../../../services/item-map/item-map.service';
import { GlobalService } from './../../../../services/global/global.service';
import { BatchMasterService } from './../../../../services/batchMaster/batch-master.service';
import { BatchMaster } from './../../../../shared/models/BatchMaster';
import { Component, Input, OnInit } from '@angular/core';
import { BatchDetailsDto } from 'src/app/shared/models/BatchDetailsDTO';
import { MATERIAL_TYPE } from 'src/app/shared/enums/batch.enum';
import { PaginationParams } from 'src/app/shared/models/paginationParams';
import { ItemMapDto } from 'src/app/shared/models/ItemMapDto';

@Component({
  selector: 'app-byproduct',
  templateUrl: './byproduct.component.html',
  styleUrls: ['./byproduct.component.css']
})
export class ByproductComponent implements OnInit {


  @Input() batchDetailsList: BatchDetailsDto[];
  @Input() isEditMode: boolean;
  @Input() batchMasterInfo: BatchMaster;
  paginationParams: PaginationParams = new PaginationParams();
  byProductListFromBatchDetails: BatchDetailsDto[];

  // DX Scroll View Config
  scrollByContent = true;
  scrollByThumb = true;
  scrollbarMode: string = 'always'; //  'onScroll' | 'onHover' | 'always' | 'never'
  byProductLists: ItemMapDto[];
  selectedByProduct: ItemMapDto;
  byProducts_filtered: ItemMapDto[];
  selecteditemCode:string;
  selectBoxLoader: boolean;

  constructor(public batchMasterService: BatchMasterService,
    public globalService: GlobalService,
    public itemMapService: ItemMapService) {
  }

  ngOnInit(): void {

    this.refreshDataGrid();
  }

  onToolbarPreparing(e) {
    e.toolbarOptions.items.unshift({
      location: 'after',
      widget: 'dxButton',
      options: {
        icon: 'refresh',
        onClick: this.refreshDataGrid.bind(this),
      },
    },
    // {
    //   location: 'before',
    //   widget: 'dxSelectBox',
    //   options: {
    //     width: 200,
    //     items: this.ingredientLists,
    //     displayExpr: 'itemDescription',
    //     valueExpr: 'itemCode',
    //     //value: this.selectedFilteredOption.text,
    //     // onValueChanged: this.onValueChanged.bind(this)
    //   },
    // }, {
    //   location: 'before',
    //   widget: 'dxButton',
    //   options: {
    //     icon: 'add',
    //     onClick: this.refreshDataGrid.bind(this),
    //   },
    // },
    );
  }
  refreshDataGrid() {
    this.getByProductfromBatch();
    this.getAllItemLineTypeWise(this.batchMasterInfo.orgId, MATERIAL_TYPE.BYPRODUCT);
    this.datagrid_DXSaveButton_Disable();
    //this.dataGrid.instance.refresh();
  }

  datagrid_DXSaveButton_Disable(){
    let dxSaveButton: any = document.getElementsByClassName('dx-datagrid-save-button')[0];

    if(dxSaveButton){
      dxSaveButton.style.display = 'none';
    }
  }

  datagrid_DXSaveButton_EnableClick(){
    let dxSaveButton: any ;
    dxSaveButton = document.getElementsByClassName('dx-datagrid-save-button')[0];
    if(dxSaveButton){
      dxSaveButton.style.display = 'none';
      dxSaveButton.click();
    }
  }
  getByProductfromBatch() {
    this.paginationParams.midFk = this.batchMasterInfo.midPk;
    this.paginationParams.materialType = MATERIAL_TYPE.BYPRODUCT;
    this.batchMasterService.getBatchDetails(this.paginationParams).subscribe(result => {
      if (result.statusCode == 200) {
        this.byProductListFromBatchDetails = result.values;
        let slNo = 1;
        this.byProductListFromBatchDetails.forEach((element) => {
          element.isNew = "False"
          element.slNo = slNo++;
        });
        console.log(this.byProductListFromBatchDetails);
      }
    }, err => {
      this.globalService.errorResponseHandler(err);
    })
  }

  getAllItemLineTypeWise(orgId, lineType) {
    this.selectBoxLoader = true;
    this.paginationParams.orgId = orgId;
    this.paginationParams.lineType = lineType;
    this.itemMapService.getAllItemList(this.paginationParams).subscribe(result => {
      if (result.statusCode == 200) {
        this.byProductLists = result.values;

        // this.ingredients_filtered = this.ingredientLists.filter(({ itemCode: id1 }) => ! this.ingredientListFromBatchDetails.some(({ itemCode: id2 }) => id2 === id1));

        // console.log(this.ingredientLists);
        // console.log(this.ingredients_filtered);
        this.filterProductList();

      }
    }, err => {
      this.globalService.errorResponseHandler(err);
      this.selectBoxLoader = false;
    })
  }

  onRowPreparing(e){
    //   if(e.rowType === "data" && e.column.dataField === "totalPcs") {
    //     e.isEditing = e.row.data.pInsType == 'SHORT_BILLET' ? false : true;
    // }
    if(e.data.isNew && e.data.isNew == "true") {
      e.isEditing =  true;
      e.isDeleting = true;
    } else {
      e.isEditing =  false;
      e.isDeleting = false;
    }
  }

  onRowRemoved(e){
    console.log('onRowRemoved',e);
    if(e.data.isNew == 'true'){
      const selectedData = this.byProductListFromBatchDetails.filter((s) => s.slNo === e.data.slNo)[0];
      let index = this.byProductListFromBatchDetails.findIndex(s => s.slNo == e.data.slNo);
      if(index != -1){
        // Found
        this.byProductListFromBatchDetails.slice(index, 1);
      }
    }
    this.filterProductList();
  }

  onRowUpdated(e){
    console.log('onRowUpdated',e);
    this.filterProductList();
    this.update([e.data]);
  }
  onCellPrepared(e){
    if(e.rowType === "data" && e.columnIndex === 5) {
      if(e.row.data.isNew == false || e.row.data.isNew.toLowerCase() == 'false') {
        //e.cellElement.childNodes[2].find(".dx-link-delete").remove();
        e.cellElement.removeChild(e.cellElement.children[1])
      }
    }
  }
  filterProductList() {
    this.byProducts_filtered = this.byProductLists.filter(({ itemCode: id1 }) => !this.byProductListFromBatchDetails.some(({ itemCode: id2 }) => id2 === id1));
    this.selectBoxLoader = false;
    // console.log(this.ingredientLists);
    // console.log(this.ingredients_filtered);
  }

  productOptionChange(e) {
    console.log(e.selectedItem);
    this.selectedByProduct = e.selectedItem;
  }

  addToList() {
    if (this.selectedByProduct) {
      let item : BatchDetailsDto = {
        slNo : 0,
        didPk: 0,
        midFk : this.batchMasterInfo.midPk.toString(),
        inventoryItemId: this.selectedByProduct.inventoryItemId,
        orgId: this.batchMasterInfo.orgId,
        l2MaterialName: this.selectedByProduct.itemDescription,
        weight:0,
        calculatedWeight: 0,
        transectedWeight: 0,
        originalQuantity: 0,
        uom: this.selectedByProduct.uom,
        materialSource: '',
        materialType: MATERIAL_TYPE.BYPRODUCT,
        yieldContribution: '',
        materialLineId: 0,
        errorMsg: '',
        isManual: '',
        createdBy: '',
        lastUpdateBy: '',
        itemCode: this.selectedByProduct.itemCode,
        itemDescription: this.selectedByProduct.itemDescription,
        creationDate: new Date(),
        lastUpdateDate: new Date(),
        isNew: "true",
        hmsAdjusted: 0,
        remarks: '',
        stockQty: 0,
      };

      // item.itemCode = this.selectedIngredient.itemCode;
      // item.itemDescription = this.selectedIngredient.itemDescription;
      // item.midFk = this.batchMasterInfo.midPk.toString();
      item.slNo = this.byProductListFromBatchDetails.length +1;
      this.byProductListFromBatchDetails.push(item)
      this.selectedByProduct = null;
      this.selecteditemCode = null;
      this.filterProductList();
    }
  }

  update(dataMdel: BatchDetailsDto[]) {
    this.datagrid_DXSaveButton_EnableClick();

    console.log(dataMdel);

    this.batchMasterService.upSertBatchDetails(dataMdel).subscribe(result => {
      if (result.statusCode == 200) {
        console.log(result);
        this.globalService.showSwal('Success', 'Successfully saved.', 'success');
        this.getByProductfromBatch()
      }
    }, err => {
      this.globalService.errorResponseHandler(err);
    })
  }
}

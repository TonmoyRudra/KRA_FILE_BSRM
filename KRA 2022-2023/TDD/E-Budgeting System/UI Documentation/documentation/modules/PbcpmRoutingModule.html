import { AssetsComponent } from './assets/assets.component';

import { CalenderAddComponent } from './calender-add/calender-add.component';

const routes: Routes = [
  {
    path: '',
    component: WelcomeComponent,
    canActivate: [AuthGuard],
  },
  {

    component: NoaAssignmentComponent,
    canActivate: [AuthGuard, PBCPMAdminGuard],
  },
  {
    path: 'userAssignment',
    component: UserSettingAssignmentComponent,
    canActivate: [AuthGuard,PBCPMAdminGuard],
  },
  {
    path: 'calender',
    component: CalenderComponent,
    canActivate: [AuthGuard,PBCPMAdminGuard],
  },
  {
    path: 'calender-add',
    component: CalenderAddComponent,
    canActivate: [AuthGuard,PBCPMAdminGuard],
  },
  {
    path: 'noaAssignmentCommonBdgt',
    component: NoaAssignmentCommonBdgtComponent,
    canActivate: [AuthGuard, PBCPMAdminGuard],
  },
  {
    path: 'init-budget',
    component: InitBudgetComponent,
    canActivate: [AuthGuard, PBCPMAdminGuard],
  },
  {
    path: 'org-setup',
    component: OrgSetupComponent,
    canActivate: [AuthGuard, PBCPMAdminGuard],
  },
  {
    path: 'common-budget-types',
    component: CommonBudgetTypeComponent,
    canActivate: [AuthGuard, PBCPMAdminGuard],
  },
  {
    path: 'asset-category',
    component: AssetCatagoryComponent,
    canActivate: [AuthGuard, PBCPMAdminGuard],
  },
  {
    path: 'asset',
    component: AssetsComponent,
    canActivate: [AuthGuard, PBCPMAdminGuard],
  },
  { path: '**', redirectTo: '', pathMatch: 'full' }
];

@NgModule({
  imports: [RouterModule.forChild(routes)],
  exports: [RouterModule]
})
export class PbcpmRoutingModule { }

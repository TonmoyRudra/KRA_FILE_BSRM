import { EbgtUserSettingDto } from './../../../../shared/models/userSettings-dto.model';
import { PbcpmService } from './../../../../services/pbcpm/pbcpm.service';
import { GlobalService } from './../../../../services/global/global.service';
import { AuthService } from './../../../../services/auth/auth.service';
import { Component, OnInit } from '@angular/core';
import { NgxSpinnerService } from 'ngx-spinner';
import { CommonSettingService } from 'src/app/services/common-setting/common-setting.service';
import ArrayStore from 'devextreme/data/array_store';
import CustomStore from 'devextreme/data/custom_store';

@Component({
  selector: 'app-user-setting-assignment',
  templateUrl: './user-setting-assignment.component.html',
  styleUrls: ['./user-setting-assignment.component.css']
})
export class UserSettingAssignmentComponent implements OnInit {

  showBulkUploader = false;
  sessionUser: any;
  title = 'User Setting Assignment With Workflow';
  sheetName = 'User_Setting_With_Workflow.xlsx';
  downloadCSVTempleteLink = this.globalService.sheetTempleteURL + this.sheetName;
  dataSource: any;
  userList: any;
  selectedCalender: any;
  popupVisible: any;
  employeeListDataSource: ArrayStore;
  selectedFromEmpId: any[];
  selectedToEmpIdId: any[];
  constructor(   private spinner: NgxSpinnerService,
    private authService: AuthService,
    private globalService: GlobalService,
    private pbcpmService: PbcpmService,
    private commonSettingService : CommonSettingService) {
    this.sessionUser = this.authService.getSessionUser();
    this.selectedCalender = this.commonSettingService.getSelectedCalender();
  }

  ngOnInit(): void {
    this.getHrEmployees();
    this.getUserList();
    this.selectedCalender = this.commonSettingService.getSelectedCalender();
  }
  generateDataGrid(list) {
    this.dataSource = new CustomStore({
      key: 'pid',
      load: () => list,
      insert: (values) => this.createUserSettingSingle(values),
      update: (key, values) => this.updateUserSettingList(key, values),
      remove: (key) => this.deleteUserSettingList(key)
    });
  }
  onToolbarPreparing(e) {
    e.toolbarOptions.items.unshift({
      location: 'after',
      widget: 'dxButton',
      options: {
        icon: 'refresh',
        onClick: this.refreshDataGrid.bind(this),
      },
    });
  }
  refreshDataGrid() {
      this.getUserList();
    //this.dataGrid.instance.refresh();
  }
  showUploader(bool: boolean) {
    this.showBulkUploader = bool;
  }

  updateUserSettingList(key, values): any {
    console.log('Update', key, values);
    var selectedData = this.userList.filter((s) => s.pid === key)[0];

    for (const key in values) {
      if (selectedData.hasOwnProperty(key)) {
        selectedData[key] = values[key];
      }
    }
    console.log('Update', selectedData);

    this.createUserAssignment([selectedData], 'update'); //

  }

  createUserSettingSingle(values): any {
    console.log('Create', values);
    this.createUserAssignment([values], 'addSingle'); //

  }

  deleteUserSettingList(key): any{
    console.log('Delete', key);
    this.deleteUserSetting(key);

  }

  async getUserList() {
    (await this.pbcpmService.getAllUserSettings()).subscribe(
      (result) => {
        if (result.statusCode == 200) {
          // this.programList = result.resultset;
         this.userList = result.result;
          let slNo = 1;
          this.userList.forEach((element) => {
            element.slNo = slNo++;
            element.costcenterCode = element.costcenterCode.padStart(4, '0');
            element.locationCode = element.locationCode.padStart(3, '0');
          });

          this.generateDataGrid(this.userList);
        }
      },
      (err) => {
        console.log(err);
      }
    );
  }

  createUserAssignment(body: any[], action='add') {
    this.spinner.show();
    let validation : boolean;
    var model =  body;
    if(action == 'add'){
      body.forEach(element => {
        element.Remarks = element.Remarks ? element.Remarks.toString() : null;
        element.CostcenterCode = element?.CostcenterCode ? element.CostcenterCode.toString() : null;
        element.LocationCode = element?.LocationCode ? element.LocationCode.toString() : null;
        element.RegionId = element?.RegionId ? element.RegionId.toString() : null;
        element.OrgId = element?.OrgId ? Number(element.OrgId) : null;
        element.ComBgtId = element?.ComBgtId ? Number(element.ComBgtId) : null;
        element.PreparerId = element?.PreparerId ? Number(element.PreparerId) : null;
        element.Approver1Id = element?.Approver1Id ? Number(element.Approver1Id) : null;
        element.Approver2Id = element?.Approver2Id ? Number(element.Approver2Id) : null;
        element.Approver3Id = element?.Approver3Id ? Number(element.Approver3Id) : null;
        element.Approver4Id = element?.Approver4Id ? Number(element.Approver4Id) : null;
        element.Approver5Id = element?.Approver5Id ? Number(element.Approver5Id) : null;
        element.Approver6Id = element?.Approver6Id ? Number(element.Approver6Id) : null;
        element.Approver7Id = element?.Approver7Id ? Number(element.Approver7Id) : null;
        element.Approver8Id = element?.Approver8Id ? Number(element.Approver8Id) : null;
        element.Approver9Id = element?.Approver9Id ? Number(element.Approver9Id) : null;
        element.Approver10Id = element?.Approver10Id ? Number(element.Approver10Id) : null;
        element.PbcpmId = element?.PbcpmId ? Number(element.PbcpmId) : null;

        // if((element.BgtTypeId == 1 || element.BgtTypeId == 2  || element.BgtTypeId == 3) && element.OrgId == null ){
        //   // Check ordId must
        //   this.spinner.hide();
        //   this.globalService.showSwal('Warning', 'Org Id mandatory for Production, RMPM and Store Budget.' , 'error');
        //   validation = false;
        // } else {
        //   validation = true;
        // }

      });
    }


    this.pbcpmService.createUserAssignment(body).subscribe(
      (result) => {
        if (result.statusCode == 200) {
          this.spinner.hide();
          console.log('create Users Done');
          console.log(result);

          if(action == 'add') this.globalService.showSwal('Success' , 'Succesfully User assignment done.', 'success')
          else if(action == 'update') this.globalService.showSwal('Update' , 'Succesfully Update.', 'success')
          else this.globalService.showSwal('Success' , 'Succesfully User assignment done.', 'success')

          this.showUploader(false);
          this.getUserList();
        }
      },
      (err) => {
        this.spinner.hide();
        this.globalService.showSwal('Warning', err.Message, 'error');

      }
    );
  }

  deleteUserSetting(id) {
    this.pbcpmService.deleteUserAssignment(id).subscribe(
      (result) => {
        if (result.statusCode == 200) {
          this.spinner.hide();
          console.log('create Users Done');
          console.log(result);

          this.globalService.showSwal('Deleted' , result.message, 'success')

          this.showUploader(false);
          this.getUserList();
        }
      },
      (err) => {
        this.spinner.hide();
        this.globalService.showSwal('Warning', err.Message, 'error');

      }
    );
  }

  async getHrEmployees() {
    //this.globalService.showSpinner(true);
    (await this.commonSettingService.getHrEmployees()).subscribe(
      (result) => {
        if (result.statusCode == 200) {
          // this.programList = result.resultset;
          this.employeeListDataSource = new ArrayStore({
            data: result.result,
            key: "hrEmployeeid",
          });
        }
      },
      (err) => {
        //this.globalService.showSpinner(false);
        console.log(err);
      }
    );
  }

  showchangeApproverPopup(bool: boolean) {
    this.selectedFromEmpId = [];
    this.selectedToEmpIdId = [];
    this.popupVisible = bool;
  }

  submitApproverPopup(e){
    console.log(this.selectedFromEmpId)
    console.log(this.selectedToEmpIdId)
    this.selectedCalender.calendarId


    if(this.selectedFromEmpId.length > 0 && this.selectedToEmpIdId.length > 0){
      this.pbcpmService.updateApprover(this.selectedFromEmpId[0], this.selectedToEmpIdId[0], this.selectedCalender.calendarId).subscribe(result => {
        if(result.statusCode == 200) {
          this.globalService.showSwal('Success','Successfully change approver', 'success')
        }
      }, err =>{
        console.log(err);
        this.globalService.errorResponseHandler(err);
      })
    }
  }
}

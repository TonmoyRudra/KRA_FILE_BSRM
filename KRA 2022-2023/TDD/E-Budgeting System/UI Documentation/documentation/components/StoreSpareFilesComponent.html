import { StoreSpareBdgtService } from './../../../../services/storeSpare-bdgt/storeSpare-bdgt.service';
import { UploadService } from './../../../../services/upload/upload.service';
import { Component, OnInit } from '@angular/core';
import { NgxSpinnerService } from 'ngx-spinner';
import { AuthService } from 'src/app/services/auth/auth.service';
import { CommonSettingService } from 'src/app/services/common-setting/common-setting.service';

import { PbcpmService } from 'src/app/services/pbcpm/pbcpm.service';

export class StoreSpareFilesComponent implements OnInit {

  sessionUser: any;
  selectedCalender: any;
  fileData: File;
  dataSource: any;
  storeFileList: any;
  fileModel: { file: File; fileId: number; comments: string; };
  fileUploadRes: any = [];
  fileUploadHttpHeader: { Authorization: string; };
  fileUrl: any;
  downloadCSVTempleteLink: any = this.globalService.sheetTempleteURL + 'STORE_File.xlsx';

  constructor(   private spinner: NgxSpinnerService,
    private authService: AuthService,
    public globalService: GlobalService,
    private storeSpareBdgtService: StoreSpareBdgtService,
    private commonSettingService : CommonSettingService,
    private uploadService: UploadService) {
    this.sessionUser = this.authService.getSessionUser();
    this.selectedCalender = this.commonSettingService.getSelectedCalender();
    this.fileUploadHttpHeader = { 'Authorization': 'Bearer ' + tokenGetter() };
  }

  ngOnInit(): void {
    this.refreshDataGrid();
  }

  onToolbarPreparing(e) {
    e.toolbarOptions.items.unshift({
      location: 'after',
      widget: 'dxButton',
      options: {
        icon: 'refresh',
        onClick: this.refreshDataGrid.bind(this),
      },
    });
  }

  refreshDataGrid() {
    this.getList();

    //this.dataGrid.instance.refresh();
  }

  async getList(){
    this.globalService.showSpinner(true);
    (await this.storeSpareBdgtService.getAllStoreFile(0)).subscribe(
      (result) => {
        if (result.statusCode == 200) {
          // this.programList = result.resultset;
          this.storeFileList= result.result;
          let slNo = 1;
          this.storeFileList.forEach((element) => {
            element.slNo = slNo++;
            element.costcenterCode = element.costcenterCode.padStart(4, '0');
            element.locationCode = element.locationCode.padStart(3, '0');
          });
          console.log(this.storeFileList)
          this.generateDataGrid(this.storeFileList);
          this.globalService.showSpinner(false);
        }
      },
      (err) => {
        this.globalService.showSpinner(false);

        console.log(err);
      }
    );
  }

  generateDataGrid(list) {
    this.dataSource = new CustomStore({
      key: 'pid',
      load: () => list,
      insert: (values) => this.createStoreFile(values),
      update: (key, updateBody) => this.updateStoreFile(key, updateBody),
      remove: (key) => this.deleteStoreFile(key)
    });
  }

  fileBrowseHandler($event) {
    //this.onFileChange($event);
  }

  createStoreFile(body: any): any {
    this.spinner.show();
    delete body.attachedRmpmMediaId;
    if(this.fileUploadRes.length > 0){
      body.attachedMediaId = Number(this.fileUploadRes[0].mid);
    }
    // Remove zero from leading '0151' > '151'
    body.costcenterCode =  body.costcenterCode.replace(/^0+/, '');
    body.locationCode =  body.locationCode.replace(/^0+/, '');
    this.storeSpareBdgtService.createStoreFile(body).subscribe(result => {
      if (result.statusCode == 200) {
        this.spinner.hide();
        this.globalService.showSwal('Create' , 'Succesfully Created.', 'success');
        this.getList();

        // clear
        this.fileUploadRes = [];
      }

    }, err => {
      console.log(err);
      this.spinner.hide();
      this.fileUploadRes = [];
      this.globalService.showSwal('Warning', err.Message, 'error');
    });
  }

  updateStoreFile(key, updateBody: any): any {

    console.log('Update', key, updateBody);
    var selectedData = this.storeFileList.filter((s) => s.pid === key)[0];

    for (const key in updateBody) {
      if (selectedData.hasOwnProperty(key)) {
        selectedData[key] = updateBody[key];
      }
    }
    console.log('Update', selectedData);
    this.spinner.show();
    //delete selectedData.attachedRmpmMediaId;
    if(this.fileUploadRes.length > 0){
      selectedData.attachedMediaId = Number(this.fileUploadRes[0].mid);
    }

    // Remove zero from leading '0151' > '151'
    //selectedData.costcenterCode =  selectedData.costcenterCode.replace(/^0+/, '');
    //selectedData.locationCode =  selectedData.locationCode.replace(/^0+/, '');
    this.storeSpareBdgtService.editStoreFile(selectedData).subscribe(result => {
      if (result.statusCode == 200) {
        this.spinner.hide();
        this.globalService.showSwal('Create' , 'Succesfully Updated.', 'success');
        this.getList();
        this.fileUploadRes = [];
      }

    }, err => {
      console.log(err);
      this.spinner.hide();
      this.fileUploadRes = [];
      this.globalService.showSwal('Warning', err.Message, 'error');
    })
  }

  deleteStoreFile(id : any) : any {
    this.storeSpareBdgtService.deleteStoreFile(id).subscribe(
      (result) => {
        if (result.statusCode == 200) {
          this.spinner.hide();
          console.log(result);

          this.globalService.showSwal('Deleted' , result.message, 'success')

          //this.showUploader(false);
          this.getList();
        }
      },
      (err) => {
        this.spinner.hide();
        this.globalService.showSwal('Warning', err.Message, 'error');

      }
    );
  }

  downloadFile(data) {
    this.uploadService.download(data.data.guidFileName).subscribe((result) =>{
      if(result){
        console.log(result);
        FileSaver.saveAs(result, data.data.originalFileName);
      }
    }, err => {
      console.log(err);
    })
  }

  onValueChanged(e: any): void {
    var reader: FileReader = new FileReader();
    // reader.onload = (args) => {
    //   if (typeof args.target.result === 'string') {
    //     //this.uploadedImageRef.src = args.target.result;
    //   }
    // };
    // reader.readAsDataURL(e.value[0]); // convert to base64 string
    reader =  e.value[0]; // file
    console.log(reader)
  }

  onUploaded(e: any, cellInfo: any): void {

    //this.retryButtonVisible = false;
    var xhttp = e.request;
    console.log(xhttp);
    if (xhttp.status != 200) //xhttp.readyState == 4 &&
    {
      console.log("Connection refused.");
      // notify("File upload error. " + xhttp.responseText, "success", 3000);
    }

    this.fileUploadRes = JSON.parse(e.request.responseText);
    cellInfo.setValue(this.fileUploadRes[0].guidFileName);
  }

  onUploadError(e: any): void {
    this.globalService.showSwal('Failed','File Uploaded Failed. Please try again later.','error')
    const xhttp = e.request;
    if (xhttp.status === 400) {
      e.message = e.error.responseText;
    }
    if (xhttp.readyState === 4 && xhttp.status === 0) {
      e.message = 'Connection refused';
    }
    //this.retryButtonVisible = true;
  }

  downloadTemplete() {
    if (this.downloadCSVTempleteLink) {
      window.open(this.downloadCSVTempleteLink, '_blank');
    } else {
      Swal.fire('Warning', 'No Download file found.', 'warning');
    }
  }
}

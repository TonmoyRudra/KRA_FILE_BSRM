import { NOAForCommonDTO } from './../../../../shared/models/noa-common-dto.model';
import { PbcpmService } from './../../../../services/pbcpm/pbcpm.service';
import { GlobalService } from './../../../../services/global/global.service';
import { AuthService } from './../../../../services/auth/auth.service';
import { Component, OnInit } from '@angular/core';
import { NgxSpinnerService } from 'ngx-spinner';
import CustomStore from 'devextreme/data/custom_store';

@Component({
  selector: 'app-noa-assignment-common-bdgt',
  templateUrl: './noa-assignment-common-bdgt.component.html',
  styleUrls: ['./noa-assignment-common-bdgt.component.css']
})
export class NoaAssignmentCommonBdgtComponent implements OnInit {

  showBulkUploader = false;
  sessionUser: any;
  catagory = 'masterdata';
  subcatagory = 'staff-mapping';
  title = 'NOA Assignment For Common Budget';
  sheetName = 'NOA_Assignment_Common_Budget.xlsx';
  downloadCSVTempleteLink = this.globalService.sheetTempleteURL + this.sheetName;
  dataSource: any;
  noalist: any;

  constructor(   private spinner: NgxSpinnerService,
    private authService: AuthService,
    private globalService: GlobalService,
    private pbcpmService: PbcpmService) {
    this.sessionUser = this.authService.getSessionUser();
  }

  ngOnInit(): void {
    this.showBulkUploader = false;
    this.getNOAList();
  }

  onToolbarPreparing(e) {
    e.toolbarOptions.items.unshift({
      location: 'after',
      widget: 'dxButton',
      options: {
        icon: 'refresh',
        onClick: this.refreshDataGrid.bind(this),
      },
    });
  }

  refreshDataGrid() {
    this.getNOAList();

    //this.dataGrid.instance.refresh();
  }
  generateDataGrid(list) {
    this.dataSource = new CustomStore({
      key: 'pid',
      load: () => list,
      insert: (values) => this.createNOAAssignmentSingle(values),
      update: (key, values) => this.updateNOAAssignment(key, values),
      remove: (key) => this.deleteNOAAssignment(key)
    });
  }

  deleteNOAAssignment(key: any):any {
    this.deleteNoaAssignment(key);
  }

  updateNOAAssignment(key: any, values: any): any{
    console.log('Update', key, values);
    var selectedData = this.noalist.filter((s) => s.pid === key)[0];

    for (const key in values) {
      if (selectedData.hasOwnProperty(key)) {
        selectedData[key] = values[key];
      }
    }
    console.log('Update', selectedData);
    this.createNOAAssignment([selectedData], 'update');
  }

  createNOAAssignmentSingle(values: any): any {
    this.createNOAAssignment([values],'addSingle');
  }

  showUploader(bool: boolean) {
    this.showBulkUploader = bool;
  }

  async getNOAList() {
    this.globalService.showSpinner(true);
    (await this.pbcpmService.getAllNOA_CommonBudget()).subscribe(
      (result) => {
        if (result.statusCode == 200) {
          // this.programList = result.resultset;
          this.noalist= result.result;
          let slNo = 1;
          this.noalist.forEach((element) => {
            element.slNo = slNo++;
            element.locationCode = element.locationCode.padStart(3, '0');
            element.legalEntityCode = element.legalEntityCode.padStart(2, '0');
          });
          console.log(this.noalist)
          this.generateDataGrid(this.noalist);
          this.globalService.showSpinner(false);
        }
      },
      (err) => {
        this.globalService.showSpinner(false);

        console.log(err);
      }
    );
  }

  createNOAAssignment(body: NOAForCommonDTO[], action='add') {
    this.spinner.show();
    var model =  body;
    if(action == 'add'){
      body.forEach(element => {
        element.CbId = Number(element.CbId);
        element.LegalEntityCode = element.LegalEntityCode.toString();
        element.CostCenterCode = element.CostCenterCode.toString();
        element.LocationCode = element.LocationCode.toString();
        element.NoaCode = element.NoaCode.toString();
        element.LastYearAmount1 = element.LastYearAmount1 == null ? 0 : element.LastYearAmount1;
        element.LastYearAmount2 = element.LastYearAmount2 == null ? 0 : element.LastYearAmount2;
        element.LastYearAmount3 = element.LastYearAmount3 == null ? 0 : element.LastYearAmount3;
      });
    }

    this.pbcpmService.createUserAssignment_CommonBudget(model).subscribe(
      (result) => {
        if (result.statusCode == 200) {
          this.spinner.hide();
          console.log('create NOA Done');
          if(action == 'update') this.globalService.showSwal('Update' , 'Succesfully Update.', 'success')
          else this.globalService.showSwal('Success' , 'Succesfully NOA assignment done.', 'success')

          console.log(result);
          this.showUploader(false);
          this.getNOAList();
        }
      },
      (err) => {
        this.spinner.hide();
        this.globalService.showSwal('Warning', err.Message, 'error');

      }
    );
  }
  deleteNoaAssignment(id) {
    this.pbcpmService.deleteUserAssignment_CommonBudget(id).subscribe(
      (result) => {
        if (result.statusCode == 200) {
          this.spinner.hide();
          console.log('create Users Done');
          console.log(result);

          this.globalService.showSwal('Deleted' , result.message, 'success')

          this.showUploader(false);
          this.getNOAList();
        }
      },
      (err) => {
        this.spinner.hide();
        this.globalService.showSwal('Warning', err.Message, 'error');

      }
    );
  }
}

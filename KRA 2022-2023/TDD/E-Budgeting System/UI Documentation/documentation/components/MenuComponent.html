  collapsiableMenuItemJS(){
    var coll = document.getElementsByClassName("collapsible");
    var i;
    for (i = 0; i < coll.length; i++) {
      coll[i].addEventListener("click", function() {
        this.parentElement.classList.toggle("active");
        //var content = this.nextElementSibling;
        //this.nextSibling.classList.toggle('show');
        if(this.nextSibling.classList.contains('show')){
          this.nextSibling.classList.remove("show");
        } else {
          this.nextSibling.classList.add("show");
        }
      });
    }
  }

  refreshDataGrid() {
    this.selectedCalender = this.commonSettingService.getSelectedCalender();
    if (this.authService.isPBCPM()) {
      this.getAllBudget(this.selectedCalender.calendarId, 0, 0, 1);
    } else {
      this.getAllBudget(this.selectedCalender.calendarId, this.sessionUser.hr_employee_id, this.bdgt_type_id, 1);
    }
  }

  async getAllBudget(calId, userId, typeId=0, actionPick: number) {

    (await this.budgetService.getBudgetListBy(calId, userId, typeId)).subscribe(
      (result) => {
        if (result.statusCode == 200) {
          // this.programList = result.resultset;
          this.dataSource = result.result;
          // if (actionPick == 2) {
          //   // Only Pending by user
          //   this.dataSource = this.dataSource.filter(x => x.statusby == this.sessionUser.user_name)
          // }
          let slNo = 1;
          this.dataSource?.forEach((element) => {
            element.slNo = slNo++;
          });
          this.typeWiseBudget = _.mapValues(_.groupBy(this.dataSource, 'budgetTypeId'),
                          clist => clist.map(car => _.omit(car, 'budgetTypeId')));

          console.log(this.typeWiseBudget);
          //this.collapsiableMenuItemJS();
          // this.globalService.showSpinner(false);
        }
      },
      (err) => {
        this.collapsiableMenuItemJS();

        //this.globalService.showSpinner(false);
        console.log(err);
      }
    );
  }
}

import { CommonBdgtService } from 'src/app/services/common-bdgt/common-bdgt.service';
import { UploadService } from './../../../../services/upload/upload.service';
import { Component, OnInit } from '@angular/core';
import { NgxSpinnerService } from 'ngx-spinner';
import { AuthService } from 'src/app/services/auth/auth.service';
import { CommonSettingService } from 'src/app/services/common-setting/common-setting.service';
import { GlobalService } from 'src/app/services/global/global.service';
import { PbcpmService } from 'src/app/services/pbcpm/pbcpm.service';
import * as XLSX from 'xlsx';
import Swal from 'sweetalert2';
import CustomStore from 'devextreme/data/custom_store';
import { tokenGetter } from 'src/app/app.module';
import * as FileSaver from 'file-saver';
@Component({
  selector: 'app-asset-catagory',
  templateUrl: './asset-catagory.component.html',
  styleUrls: ['./asset-catagory.component.css']
})
export class AssetCatagoryComponent implements OnInit {

  sessionUser: any;
  selectedCalender: any;
  fileData: File;
  dataSource: CustomStore;
  assetCategoryList: any;
  fileModel: { file: File; fileId: number; comments: string; };
  fileUploadRes: any = [];
  fileUploadHttpHeader: { Authorization: string; };
  fileUrl: any;

  constructor(   private spinner: NgxSpinnerService,
    private authService: AuthService,
    public globalService: GlobalService,
    private commonBdgtService: CommonBdgtService,
    private commonSettingService : CommonSettingService,
    private uploadService: UploadService) {
    this.sessionUser = this.authService.getSessionUser();
    this.selectedCalender = this.commonSettingService.getSelectedCalender();
    this.fileUploadHttpHeader = { 'Authorization': 'Bearer ' + tokenGetter() };
  }
  ngOnInit(): void {
    this.refreshDataGrid();
  }

  onToolbarPreparing(e) {
    e.toolbarOptions.items.unshift({
      location: 'after',
      widget: 'dxButton',
      options: {
        icon: 'refresh',
        onClick: this.refreshDataGrid.bind(this),
      },
    });
  }

  refreshDataGrid() {
    this.getAssetCategoryList();

    //this.dataGrid.instance.refresh();
  }

  async getAssetCategoryList(){
    this.globalService.showSpinner(true);
    (await this.commonSettingService.getAssetCategoryList()).subscribe(
      (result) => {
        if (result.statusCode == 200) {
          // this.programList = result.resultset;
          this.assetCategoryList= result.result;
          let slNo = 1;
          this.assetCategoryList.forEach((element) => {
            element.slNo = slNo++;
          });
          console.log(this.assetCategoryList)
          this.generateDataGrid(this.assetCategoryList);
          this.globalService.showSpinner(false);
        }
      },
      (err) => {
        this.globalService.showSpinner(false);

        console.log(err);
      }
    );
  }

  generateDataGrid(list) {
    this.dataSource = new CustomStore({
      key: 'pid',
      load: () => list,
      insert: (values) => this.create(values),
      update: (key, updateBody) => this.update(key, updateBody),
      remove: (key) => this.delete(key)
    });
  }

  create(body: any): any {
    this.spinner.show();
    body.pid = 0; //default
    this.commonSettingService.createAssetCategory(body).subscribe(result => {
      if (result.statusCode == 200) {
        this.spinner.hide();
        this.globalService.showSwal('Create' , 'Succesfully Created.', 'success');
        this.getAssetCategoryList();
      }

    }, err => {
      console.log(err);
      this.spinner.hide();
      this.globalService.showSwal('Warning', err.Message, 'error');
    })
  }

  update(key, updateBody: any): any {

    console.log('Update', key, updateBody);
    var selectedData = this.assetCategoryList.filter((s) => s.pid === key)[0];
    // this.dataSource.byKey(key).then((dataItem) => {
    //   selectedData = dataItem;

    // }, (error) => {
    // selectedData = [];

    // })

    for (const key in updateBody) {
      if (selectedData.hasOwnProperty(key)) {
        selectedData[key] = updateBody[key];
      }
    }
    console.log('Update', selectedData);
    this.spinner.show();
    this.commonSettingService.updateAssetCategory(selectedData).subscribe(result => {
      if (result.statusCode == 200) {
        this.spinner.hide();
        this.globalService.showSwal('Create' , 'Succesfully Updated.', 'success');
        this.getAssetCategoryList();
      }

    }, err => {
      console.log(err);
      this.spinner.hide();
      this.globalService.showSwal('Warning', err.Message, 'error');
    })
  }

  delete(id : any) : any {
    this.commonSettingService.deleteAssetCategory(id).subscribe(
      (result) => {
        if (result.statusCode == 200) {
          this.spinner.hide();
          console.log(result);

          this.globalService.showSwal('Deleted' , result.message, 'success')

          //this.showUploader(false);
          this.getAssetCategoryList();
        }
      },
      (err) => {
        this.spinner.hide();
        this.globalService.showSwal('Warning', err.Message, 'error');

      }
    );
  }
}

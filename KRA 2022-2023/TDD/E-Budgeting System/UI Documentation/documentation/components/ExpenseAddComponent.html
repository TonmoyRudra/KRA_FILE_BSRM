  ngOnInit(): void {

  onToolbarPreparing(e) {
    e.toolbarOptions.items.unshift({

  refreshDataGrid() {
    this.getBudgetMasterInfo(this.budget_id);
    this.getexprenceBudgetById(this.budget_id);

  getCalenderDetailsByCalID(calId) {
    this.commonSettingService.getCalendarById(calId).subscribe(
      (result) => {
        if (result.statusCode == 200) {
          // this.programList = result.resultset;
          //this.generateGridsource(result.result);
          this.selectedBudgetCalDetails = result.result;
          this.fiscalYearList = this.getFiscalYear(
            this.selectedBudgetCalDetails.fromDate
          );
        }
      },
      (err) => {
        this.globalService.showSpinner(false);
        console.log(err);
      }
    );
  }
  async getBudgetMasterInfo(bdgtId) {
    this.globalService.showSpinner(true);
    (await this.budgetService.getBudgetMasterInfoBy(bdgtId)).subscribe(
      (result) => {
        if (result.statusCode == 200) {
          // this.programList = result.resultset;
          this.budgetMasterInfo = result.result;
          console.log(this.budgetMasterInfo);
          this.getCalenderDetailsByCalID(this.budgetMasterInfo.calendarId);
          this.userRoleForThisBudget =
            this.budgetService.getUserRoleForThisBudget(this.budgetMasterInfo);
          console.log('role', this.userRoleForThisBudget);
          this.editingPermissionGenerate();
          this.globalService.showSpinner(false);

          element.style.height = '450px';

  editingPermissionGenerate() {
    console.log(
      'editing',
      this.userRoleForThisBudget.roleName,
      this.userRoleForThisBudget.statusId
    );
    this.allowUpdating = false;
    this.editCellForPBCPM = false;
    this.editCellForPreparer = false;

    if (
      this.userRoleForThisBudget.roleName == 'preparer' &&
      this.userRoleForThisBudget.statusId == 0
    ) {
      this.allowUpdating = true;
      this.editCellForPreparer = true;
    } else if (
      this.userRoleForThisBudget.roleName == 'pbcpm' &&
      this.userRoleForThisBudget.statusId == 11
    ) {
      this.allowUpdating = true;
      this.editCellForPBCPM = true;
    }
  }

  generateCompanyWiseTotal() {
    this.companyWiseTotal = [];
    console.log('generateCompanyWiseTotal - ', this.dataSource);
    var groupBy = this.groupBy(this.dataSource, 'company');
    console.log(groupBy);
    var obj = { companyName: '', total: 0 };

    for (let i = 0; i < Object.keys(groupBy).length; i++) {
      const elementKey = Object.keys(groupBy)[i];
      const elementVlaues: any = Object.values(groupBy)[i];

      var obj = { companyName: '', total: 0 };
      obj.companyName = elementKey;
      elementVlaues.forEach((values) => {
        obj.total = obj.total + values.forecastAmt;
      });
      this.companyWiseTotal.push(obj);
      console.log('companyWiseTotal:', this.companyWiseTotal);
    }
  }

  groupBy(arr, criteria) {
    const newObj = arr.reduce(function (acc, currentValue) {
      if (!acc[currentValue[criteria]]) {
        acc[currentValue[criteria]] = [];
      }
      acc[currentValue[criteria]].push(currentValue);
      return acc;
    }, {});
    return newObj;
  }

      } else {
        saveModel.push(d);
      }
    });

    this.dataSource = saveModel;
    console.log('rows', rows);
    console.log(this.dataSource);
    console.log(saveModel);

    // check description mendatory when any amount found
    // Validation check Start
    this.failedList = [];
    saveModel.forEach((element) => {
      if (element.forecastAmt > 0) {
        // check description is present or not.
        if (!element.remarks) {
          this.failedList.push(element);
        }
      }
    });

    if (this.failedList.length > 0) {
      this.globalService.showSwal(
        'Required',
        'Must fill in description which have forcast amount.',
        'warning'
      );
      return;
    }
    // Validation check End
    // check description mendatory when any amount found;

    this.budgetMasterInfo = this.makeComments(this.comments);
    let expenseModel: ExpenseDto = {
      ebgtV2BflowMasterTs: this.budgetMasterInfo,
      ebgtV2ExpBgtDetailsTs: saveModel,
      isPartial: isPartial,
      action: action,
    };

    if (!isPartial) {
      Swal.fire({
        title: 'Are you sure?',
        text: 'You want to Save and Send it to next level?',
        icon: 'warning',
        showCancelButton: true,
        confirmButtonColor: '#3085d6',
        cancelButtonColor: '#d33',

        if (result.isConfirmed) {
          this.saveExpenseBudget(expenseModel);
        }
      });
    } else {
      this.saveExpenseBudget(expenseModel);
    }
  }

  getFiscalYear(currentCalFromDate: string) {
    var fiscalyearFrom;
    var fiscalyearTo;
    var fiscalyear;
    var last1;
    var last2;
    var last3;
    var today = new Date(currentCalFromDate);

    if (today.getMonth() + 1 <= 6) {
      fiscalyearFrom = today.getFullYear() - 1;
      fiscalyearTo = today.getFullYear();
      fiscalyear = fiscalyearFrom + '-' + fiscalyearTo;

      last1 = fiscalyearFrom - 1 + '-' + (fiscalyearTo - 1);
      last2 = fiscalyearFrom - 2 + '-' + (fiscalyearTo - 2);
      last3 = fiscalyearFrom - 3 + '-' + (fiscalyearTo - 3);

      // this.filterStartDate = '01-Jul-' + (today.getFullYear() - 1);
      // this.filterEndDate = new Date();

      // //fiscal year:
      // this.fiscalYearStartDate = '01-Jul-' + (today.getFullYear() - 1);
      // this.fiscalYearEndDate = '30-Jun-' + (today.getFullYear());
    } else {
      fiscalyearFrom = today.getFullYear();
      fiscalyearTo = today.getFullYear() + 1;
      fiscalyear = fiscalyearFrom + '-' + fiscalyearTo;

      last1 = fiscalyearFrom - 1 + '-' + (fiscalyearTo - 1);
      last2 = fiscalyearFrom - 2 + '-' + (fiscalyearTo - 2);
      last3 = fiscalyearFrom - 3 + '-' + (fiscalyearTo - 3);
      // this.filterStartDate = '01-Jul-' + (today.getFullYear());
      // this.filterEndDate = new Date();

      // //fiscal year:
      // this.fiscalYearStartDate = '01-Jul-' + (today.getFullYear());
      // this.fiscalYearEndDate = '30-Jun-' + (today.getFullYear() + 1);
    }
    return {
      fiscalyear: fiscalyear,
      lastYear1: last1,
      lastYear2: last2,
      lastYear3: last3,
    };
  }

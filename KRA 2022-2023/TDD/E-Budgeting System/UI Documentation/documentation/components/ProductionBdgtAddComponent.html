    this.getBudgetMasterInfo(this.budget_id);
    this.getProductionBudgetById(this.budget_id);
    this.budgetService.setWorkFlowUILoadObs(true);

  getCalenderDetailsByCalID(calId) {
     this.commonSettingService.getCalendarById(calId).subscribe(
      (result) => {
        if (result.statusCode == 200) {
          // this.programList = result.resultset;
          //this.generateGridsource(result.result);
          this.selectedBudgetCalDetails =  result.result;

        this.globalService.showSpinner(false);

      }

  async getBudgetMasterInfo(bdgtId) {
    this.globalService.showSpinner(true);
    (await this.budgetService.getBudgetMasterInfoBy(bdgtId)).subscribe(
      (result) => {
        if (result.statusCode == 200) {
          // this.programList = result.resultset;
          this.budgetMasterInfo = result.result;
          console.log(this.budgetMasterInfo);
          this.getCalenderDetailsByCalID(this.budgetMasterInfo.calendarId);
          this.userRoleForThisBudget = this.budgetService.getUserRoleForThisBudget(this.budgetMasterInfo);
          console.log('role',this.userRoleForThisBudget)
          this.editingPermissionGenerate();
          this.globalService.showSpinner(false);
          this.getItemRatio(this.budget_id, this.budgetMasterInfo.orgId);
          const element :any  =document.getElementsByClassName("dx-datagrid-rowsview")
          element.style.height = "450px";
        }
      },
      (err) => {
        this.globalService.showSpinner(false);
        console.log(err);
      }
    );
  }
  async getProductionBudgetById(bdgtId) {
    this.globalService.showSpinner(true);
    (await this.productionBdgtService.getProductionBdgtDetailsById(bdgtId)).subscribe(
      (result) => {
        if (result.statusCode == 200) {
          // this.programList = result.resultset;

           this.dataSource = result.result;

        this.globalService.showSpinner(false);
        console.log(err);
      }
    );
  }

      this.editCellForPreparer = true;

      this.editCellForPBCPM = true;
    }

  forecastAmountchange(e, cell) {
    //dx-datagrid-headers

  generateCompanyWiseTotal(){
    this.companyWiseTotal = [];
    console.log("generateCompanyWiseTotal - ", this.dataSource);
    var groupBy = this.groupBy(
      this.dataSource,
      'company'
    );
    console.log(groupBy);
    var obj = {companyName: '', total: 0};

    for (let i = 0; i < Object.keys(groupBy).length; i++) {
      const elementKey = Object.keys(groupBy)[i];
      const elementVlaues: any = Object.values(groupBy)[i];

      var obj = {companyName: '', total: 0};
      obj.companyName = elementKey;
      elementVlaues.forEach(values => {
        obj.total = obj.total + values.forecastAmt
      });
      this.companyWiseTotal.push(obj)
      console.log('companyWiseTotal:',this.companyWiseTotal);
    }
  }

  groupBy(arr, criteria) {
    const newObj = arr.reduce(function (acc, currentValue) {
      if (!acc[currentValue[criteria]]) {
        acc[currentValue[criteria]] = [];
      }
      acc[currentValue[criteria]].push(currentValue);
      return acc;
    }, {});
    return newObj;
  }

  getFiscalYear(currentCalFromDate: string) {
    var fiscalyearFrom;
    var fiscalyearTo;
    var fiscalyear;
    var last1;
    var last2;
    var last3;
    var today = new Date(currentCalFromDate);

    if ((today.getMonth() + 1) <= 6) {
      fiscalyearFrom = (today.getFullYear() - 1);
      fiscalyearTo = (today.getFullYear());
      fiscalyear = fiscalyearFrom + "-" + fiscalyearTo;

      last1 = (fiscalyearFrom - 1) + "-" + (fiscalyearTo - 1);
      last2 = (fiscalyearFrom - 2) + "-" + (fiscalyearTo - 2);
      last3 = (fiscalyearFrom - 3) + "-" + (fiscalyearTo - 3);

      // this.filterStartDate = '01-Jul-' + (today.getFullYear() - 1);
      // this.filterEndDate = new Date();

      // this.fiscalYearStartDate = '01-Jul-' + (today.getFullYear() - 1);

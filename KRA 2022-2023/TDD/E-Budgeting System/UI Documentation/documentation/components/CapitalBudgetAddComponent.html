        this.editMode = true;

      }

    e.toolbarOptions.items.unshift({

      options: {
        icon: 'refresh',

      },

  async refreshDataGrid() {
    await this.getAssetList();
    await this.getCarryForwardTypes();
    this.getCompanyList();
    await this.getAssetCategoryList();
    await this.getBudgetMasterInfo(this.budget_id);
    this.getCapitalBudgetById(this.budget_id);
    // This is for get production Budget list form all budget
    //await this.getAllBudget(this.selectedCalender.calendarId, this.sessionUser.hr_employee_id, 0, 1);

    this.budgetService.setWorkFlowUILoadObs(true);
    this.comments = null;


  }

  getCalenderDetailsByCalID(calId) {
    this.commonSettingService.getCalendarById(calId).subscribe(
      (result) => {
        if (result.statusCode == 200) {
          // this.programList = result.resultset;
          //this.generateGridsource(result.result);
          this.selectedBudgetCalDetails = result.result;
          this.fiscalYearList = this.getFiscalYear(this.selectedBudgetCalDetails.fromDate);
        }
      },
      (err) => {
        this.globalService.showSpinner(false);
        console.log(err);
      }
    );
  }

  async getBudgetMasterInfo(bdgtId) {
    this.globalService.showSpinner(true);
    (await this.budgetService.getBudgetMasterInfoBy(bdgtId)).subscribe(
      (result) => {
        if (result.statusCode == 200) {

          // this.programList = result.resultset;
          this.budgetMasterInfo = result.result;
          console.log(this.budgetMasterInfo);
          this.getCompanyList();
          this.getCalenderDetailsByCalID(this.budgetMasterInfo.calendarId);
          this.userRoleForThisBudget = this.budgetService.getUserRoleForThisBudget(this.budgetMasterInfo);
          console.log('role', this.userRoleForThisBudget)
          this.editingPermissionGenerate();
          this.globalService.showSpinner(false);
          // const element :any  =document.getElementsByClassName("dx-datagrid-rowsview")
          // element.style.height = "450px";

          //get org details
          this.getOrgList(this.budgetMasterInfo.orgId)
          return this.budgetMasterInfo;
        }
      },
      (err) => {
        this.globalService.showSpinner(false);
        console.log(err);
        return null;
      }
    );
  }

  editingPermissionGenerate() {
    this.allowUpdating = false;
    this.editCellForPBCPM = false;
    this.editCellForPreparer = false;

    if (this.userRoleForThisBudget.roleName == 'preparer' && this.userRoleForThisBudget.statusId == 0) {
      this.allowUpdating = true;
      this.editCellForPreparer = true;
    }
    if (this.userRoleForThisBudget.roleName == 'pbcpm' && this.userRoleForThisBudget.statusId == 11) {
      this.allowUpdating = true;
      this.editCellForPBCPM = true;
    }
  }

    let dxSaveButton: any = document.getElementsByClassName('dx-datagrid-save-button')[0];

    if (dxSaveButton) {
      dxSaveButton.style.display = 'none';
    }
  }

  getFiscalYear(currentCalFromDate: string) {
    var fiscalyearFrom;
    var fiscalyearTo;
    var fiscalyear;
    var last1;
    var last2;
    var last3;
    var today = new Date(currentCalFromDate);

    if ((today.getMonth() + 1) <= 6) {
      fiscalyearFrom = (today.getFullYear() - 1);
      fiscalyearTo = (today.getFullYear());
      fiscalyear = fiscalyearFrom + "-" + fiscalyearTo;

  saveBudget(dto: CapitalBdgtDto) {
    this.globalService.showSpinner(true);
    //const uuidRegexExp = /^[0-9a-fA-F]{8}\b-[0-9a-fA-F]{4}\b-[0-9a-fA-F]{4}\b-[0-9a-fA-F]{4}\b-[0-9a-fA-F]{12}$/gi;
    dto.ebgtV2CapitalDetailTs.forEach(element => {
      element.bgtId =  Number(this.budget_id);
      element.legalEntityCode = element?.legalEntityCode?.toString();
      element.assetCategoryCode = element?.assetCategoryCode?.toString();
      element.assetNameCode = element?.assetNameCode?.toString();
      element.noaCode = element?.noaCode ? element?.noaCode : '';

      if(this.globalService.checkIfValidUUID(element?.pid?.toString())){
        // Note: When user add new row, PID comes with a UUID.
        // Check this PID is UUID or Not. If UUID then remove it from model. Database Create it automatically.
        // Else nothing to do.

        delete element.pid;
      }

    });
    this.capitalBdgtService.updateCapitalBudget(dto).subscribe((result) => {
      if (result.statusCode == 200) {
        //this.dataSource = result.result;
        console.log(this.dataSource);
        this.datagrid_DXSaveButton_Disable();
        if (dto.action == this.globalService.actions.draft) {
          this.globalService.showSwal('SUCCESS', 'Your Draft submission has been saved.', 'success')
        } else if (dto.action == this.globalService.actions.review) {
          this.globalService.showSwal('SUCCESS', 'Your Review submission has been saved.', 'success')
        } else {
          this.globalService.showSwal('SUCCESS', 'Your submission has been saved.', 'success')
        }
        this.globalService.showSpinner(false);

        this.refreshDataGrid();
      }
    }, err => {
      console.log(err)
      if(err.status == 400){
        this.globalService.showSwal('ERROR','Validation error. Company, Asset Category, Asset Name is required. Please check all of this field.','error');
      } else {
        this.globalService.showSwal('ERROR', err.error.message, 'error');
      }
      this.globalService.showSpinner(false);
    })
  }

  generateCompanyWiseTotal(){
    this.companyWiseTotal = [];
    console.log("generateCompanyWiseTotal - ", this.dataSource);
    var groupBy = this.groupBy(
      this.dataSource,
      'legalEntityCode'
    );
    console.log(groupBy);
    var obj = {companyName: '', total: 0};

    for (let i = 0; i < Object.keys(groupBy).length; i++) {
      const elementKey = Object.keys(groupBy)[i];
      const elementVlaues: any = Object.values(groupBy)[i];

      var obj = {companyName: '', total: 0};
      obj.companyName = elementKey;
      elementVlaues.forEach(values => {
        obj.total = obj.total + values.amount
      });
      this.companyWiseTotal.push(obj)
      console.log('companyWiseTotal:',this.companyWiseTotal);
    }
  }

  groupBy(arr, criteria) {
    const newObj = arr.reduce(function (acc, currentValue) {
      if (!acc[currentValue[criteria]]) {
        acc[currentValue[criteria]] = [];
      }
      acc[currentValue[criteria]].push(currentValue);
      return acc;
    }, {});
    return newObj;
  }

  ngOnInit(): void {
    const that = this;
    this.popUpTitle = "Workflow for Budget - " + this.budgetId;
    console.log('app-action-workflow')
    this.load();
    // For reload the UI with any action.
    this.budgetService
      .getWorkFlowUILoadObs()
      .pipe(takeUntil(this.unsubscribe$))
      .subscribe((result) => {
        if (result) {
          this.load();
        }
      });

    this.saveButtonOptions = {
      icon: '',
      text: 'Save',
      onClick(e) {

      },
    };
    this.closeButtonOptions = {
      text: 'Close',
      onClick(e) {
        that.popupVisible = false;
      },
    };
  }

  load() {
    this.getBudgetWorkflowInfo(this.budgetId);
    this.getBudgetLogHistoryInfo(this.budgetId);
    this.getHrEmployees();
  }
  ngOnDestroy() {
    this.unsubscribe$.next(true);
    this.unsubscribe$.complete();
  }

        if (result.statusCode == 200) {
          // this.programList = result.resultset;
          this.budgetWorkflowInfo = result.result;
          console.log(this.budgetWorkflowInfo);

          let dataArray: workflowModel[] = [];

          if (this.budgetWorkflowInfo.preparerId > 0) dataArray.push({ flow_id: '0', flow_name: 'Preparer', employee_id: this.budgetWorkflowInfo.preparerId, employee_name: this.budgetWorkflowInfo.preparerName, employee_cmnt: this.budgetWorkflowInfo.preparerCmt, employee_details: this.budgetWorkflowInfo.preparerDetails })
          if (this.budgetWorkflowInfo.approver1Id > 0) dataArray.push({ flow_id: '1', flow_name: 'Approver 1', employee_id: this.budgetWorkflowInfo.approver1Id, employee_name: this.budgetWorkflowInfo.approver1Name, employee_cmnt: this.budgetWorkflowInfo.approver1Cmt, employee_details: this.budgetWorkflowInfo.approver1Details })

  workFlowEditPopUp() {

  nextStepForward() {
    console.log('next step move');

    Swal.fire({
      title: 'Are you sure?',
      text: 'You want to move to the next level?',
      icon: 'warning',
      showCancelButton: true,
      cancelButtonText: 'No',
      confirmButtonColor: '#3085d6',
      cancelButtonColor: '#d33',
      confirmButtonText: 'Yes, Do it!',
    }).then((result) => {
      /* Read more about isConfirmed, isDenied below */
      if (result.isConfirmed) {
        this.globalService.showSpinner(true);
        this.budgetService.workflowStepChange(this.budgetId, 'next', -1).subscribe(result => {
          if (result.statusCode == 200) {
            console.log(result.result);
            this.globalService.showSpinner(false);

            this.budgetService.setWorkFlowUILoadObs(true);
            this.globalService.showSwal('Success', 'Succesfully Move the workflow to the next approver for budget: ' + this.budgetId, 'success');

          }

        }, err => {
          this.globalService.showSpinner(false);

          console.log(err);
        })
      }
    });

  }

  userChanged(data, userRole) {
    // console.log('dropdown data changed');
    // console.log(data);
    if (data.value == null) {
      if (userRole == 0) this.selectedPreparerId = [];
      if (userRole == 1) this.selectedApprover1Id = [];
      if (userRole == 2) this.selectedApprover2Id = [];
      if (userRole == 3) this.selectedApprover3Id = [];
      if (userRole == 4) this.selectedApprover4Id = [];
      if (userRole == 5) this.selectedApprover5Id = [];
      if (userRole == 6) this.selectedApprover6Id = [];
      if (userRole == 7) this.selectedApprover7Id = [];
      if (userRole == 8) this.selectedApprover8Id = [];
      if (userRole == 9) this.selectedApprover9Id = [];
      if (userRole == 10) this.selectedApprover10Id = [];
      if (userRole == 11) this.selectedPbcpmId = [];
    }
    if (userRole == 0) this.budgetWorkflowInfo.preparerId = this.selectedPreparerId[0];
    if (userRole == 1) this.budgetWorkflowInfo.approver1Id = this.selectedApprover1Id[0];
    if (userRole == 2) this.budgetWorkflowInfo.approver2Id = this.selectedApprover2Id[0];
    if (userRole == 3) this.budgetWorkflowInfo.approver3Id = this.selectedApprover3Id[0];
    if (userRole == 4) this.budgetWorkflowInfo.approver4Id = this.selectedApprover4Id[0];
    if (userRole == 5) this.budgetWorkflowInfo.approver5Id = this.selectedApprover5Id[0];
    if (userRole == 6) this.budgetWorkflowInfo.approver6Id = this.selectedApprover6Id[0];
    if (userRole == 7) this.budgetWorkflowInfo.approver7Id = this.selectedApprover7Id[0];
    if (userRole == 8) this.budgetWorkflowInfo.approver8Id = this.selectedApprover8Id[0];
    if (userRole == 9) this.budgetWorkflowInfo.approver9Id = this.selectedApprover9Id[0];
    if (userRole == 10) this.budgetWorkflowInfo.approver10Id = this.selectedApprover10Id[0];
    if (userRole == 11) this.budgetWorkflowInfo.pbcpmId = this.selectedPbcpmId[0];

    console.log(this.budgetWorkflowInfo)
  }

  updateWorkflow() {
    this.globalService.showSpinner(true);
    console.log(this.budgetWorkflowInfo);
    var model: WorkFlow = {
      bgtId: this.budgetWorkflowInfo.budgetId,
      preparerId: this.budgetWorkflowInfo.preparerId,
      approver1Id: this.budgetWorkflowInfo.approver1Id,
      approver2Id: this.budgetWorkflowInfo.approver2Id,
      approver3Id: this.budgetWorkflowInfo.approver3Id,
      approver4Id: this.budgetWorkflowInfo.approver4Id,
      approver5Id: this.budgetWorkflowInfo.approver5Id,
      approver6Id: this.budgetWorkflowInfo.approver6Id,
      approver7Id: this.budgetWorkflowInfo.approver7Id,
      approver8Id: this.budgetWorkflowInfo.approver8Id,
      approver9Id: this.budgetWorkflowInfo.approver9Id,
      approver10Id: this.budgetWorkflowInfo.approver10Id,
      pbcpmId: this.budgetWorkflowInfo.pbcpmId
    }


    this.budgetService.updateWorkflow(model).subscribe(result => {
      if (result.statusCode == 200) {
        this.globalService.showSpinner(false);
        this.popupVisible = false;
        this.budgetService.setWorkFlowUILoadObs(true);
        // this.programList = result.resultset;
        this.globalService.showSwal('Success', 'Succesfully Update the workflow for budget: ' + this.budgetId, 'success');

      }
    }, err => {
      console.log(err)
      this.globalService.showSpinner(false);
    })
  }
}

import { ICalendar } from './../../../../shared/models/calendar';
import { CommonSettingService } from './../../../../services/common-setting/common-setting.service';
import { Component, OnInit } from '@angular/core';
import { NgxSpinnerService } from 'ngx-spinner';
import { AuthService } from 'src/app/services/auth/auth.service';
import { GlobalService } from 'src/app/services/global/global.service';
import { PbcpmService } from 'src/app/services/pbcpm/pbcpm.service';
import { Router } from '@angular/router';
import Swal from 'sweetalert2';

@Component({
  selector: 'app-calender',
  templateUrl: './calender.component.html',
  styleUrls: ['./calender.component.css']
})
export class CalenderComponent implements OnInit {
  showBulkUploader = false;
  sessionUser: any;
  dataSource: any;

  constructor(   private spinner: NgxSpinnerService,
    private authService: AuthService,
    public router: Router,
    public globalService: GlobalService,
    private pbcpmService: PbcpmService,
    private commonSettingService: CommonSettingService) {
    this.sessionUser = this.authService.getSessionUser();
  }

  ngOnInit(): void {
    this.refreshDataGrid();
  }
  onToolbarPreparing(e) {
    e.toolbarOptions.items.unshift({
      location: 'after',
      widget: 'dxButton',
      options: {
        icon: 'refresh',
        onClick: this.refreshDataGrid.bind(this),
      },
    });
  }
  refreshDataGrid() {
    this.getAll(0); // 0 = all

    //this.dataGrid.instance.refresh();
  }

  async getAll(calId) {
    (await this.commonSettingService.getAllCalender(calId)).subscribe(
      (result) => {
        if (result.statusCode == 200) {
          // this.programList = result.resultset;
          this.dataSource = result.result.sort(function(a, b){return b.calendarId-a.calendarId});
          let slNo = 1;
          this.dataSource.forEach((element) => {
            element.slNo = slNo++;
          });
        }
        else {
          // for statusCode Other
          this.globalService.showSwal(
            'Error!!',
            result.message,
            'error'
          );
        }
      },
      (err) => {
        console.log(err);
      }
    );
  }
  activeModeChange(data, e) {
    console.log(data);
    data.inactive = data.inactive == 'Y' ? 'N' : 'Y';
    // let calenderID = data.calendarId;
    // delete data.calendarId;
    this.commonSettingService.updateCalender(data).subscribe(
      (result) => {
        if(result.statusCode == 200){
          console.log(result.result);
         // this.globalService.showSwal('SUCCESS', 'Successfully update a calender', 'success');
        }
        else {
          // for statusCode Other
          this.globalService.showSwal(
            'Error!!',
            result.message,
            'error'
          );
        }
      },
      (err) => {
        console.log(err);
      });
  }

  update(data) {
    this.router.navigateByUrl(
      'main/pbcpm/calender-add?calendarId=' +
        data.calendarId
  );}

  delete(data) {
    Swal.fire({
      title: 'Are you sure?',
      text: 'You want to delete!',
      icon: 'question',
      showCancelButton: true,
      cancelButtonText: 'No',
      confirmButtonColor: '#3085d6',
      cancelButtonColor: '#d33',
      confirmButtonText: 'Yes, delete it!',
    }).then((result) => {
      /* Read more about isConfirmed, isDenied below */
      if (result.isConfirmed) {
        this.commonSettingService.deleteCalender(data.calendarId).subscribe(
          (result) => {
            console.log(result);
            if (result.statusCode == 200) {
                this.refreshDataGrid();
                this.globalService.showSwal(
                  'Deleted!!',
                  result.message,
                  'success'
                );
            } else {
              // for statusCode Other
              this.globalService.showSwal(
                'Error!!',
                result.message,
                'error'
              );
            }
          },
          (err) => {
            console.log(err);
            this.globalService.showSwal(
              'Error!!',
               err.error.message,
              'error'
            );
          }
        );
      }
    });
  }
    //this.globalService.showSwal('Will be add soon!!', 'This feature are not implement yet.', 'info')
}

import { BudgetService } from './../../../../services/budget/budget.service';
import { CommonSettingService } from './../../../../services/common-setting/common-setting.service';
import { Component, Input, OnInit,OnDestroy  } from '@angular/core';
import { NgxSpinnerService } from 'ngx-spinner';
import { AuthService } from 'src/app/services/auth/auth.service';
import { GlobalService } from 'src/app/services/global/global.service';
import { PbcpmService } from 'src/app/services/pbcpm/pbcpm.service';
import { ActivatedRoute, NavigationEnd, Router } from '@angular/router';
import Swal from 'sweetalert2';
@Component({
  selector: 'app-budget-list',
  templateUrl: './budget-list.component.html',
  styleUrls: ['./budget-list.component.css']
})
export class BudgetListComponent implements OnInit , OnDestroy {
  @Input() bdgt_type_id;

  sessionUser: any;
  dataSource: any = [];
  selectedCalender: any;
  actionList: { id: number; name: string; }[];
  actionPick: any;
  isPBCPMRole: any;

  constructor(private spinner: NgxSpinnerService,
    private authService: AuthService,
    public globalService: GlobalService,
    private pbcpmService: PbcpmService,
    public router: Router,
    public route: ActivatedRoute,
    private commonSettingService: CommonSettingService,
    private budgetService: BudgetService) {
    this.sessionUser = this.authService.getSessionUser();

    // This is for Query param change reload.
    // When we change any query param it should be reload.
    this.router.routeReuseStrategy.shouldReuseRoute = () => false;
    // this.router.routeReuseStrategy.shouldReuseRoute = function () {
    //   return false;
    // };

    // this.router.events.subscribe((evt) => {
    //   if (evt instanceof NavigationEnd) {
    //     // trick the Router into believing it's last link wasn't previously loaded
    //     this.router.navigated = false;
    //     // if you need to scroll back to top, here is the right place
    //     window.scrollTo(0, 0);
    //   }
    // });

    this.isPBCPMRole = this.authService.isPBCPM();
    this.actionList = [{ id: 1, name: 'All' }, { id: 2, name: 'Pending on me' }]

  }


  ngOnInit(): void {
    if (!this.bdgt_type_id || this.bdgt_type_id == null) { // if its not come from @Input param then check query param.
      this.route.queryParams.subscribe((params) => {
        this.bdgt_type_id = params['type'];
        if (!this.bdgt_type_id || this.bdgt_type_id == null) { // this is for PBCPM Team,
          this.bdgt_type_id = 0; // type 0 means ALl type.
        }
      });
    };

    this.selectedCalender = this.commonSettingService.getSelectedCalender();
    if(this.isPBCPMRole){
      this.actionPick = 1; // For filter dropdown ALl
    } else {
      this.actionPick = 2; // For filter dropdown Pending on me
    }
    this.refreshDataGrid();
  }

  ngOnDestroy(){
    //this.router.routeReuseStrategy.shouldReuseRoute = () => true;
    console.log("destroying child...")
  }

  makeTitle() {
    if (this.bdgt_type_id == 0) return 'ALL';
    if (this.bdgt_type_id == 1) return 'Production Budget';
    if (this.bdgt_type_id == 2) return 'Raw Packing Consumable Consumption Budget';
    if (this.bdgt_type_id == 3) return 'Store and Spare Budget';
    if (this.bdgt_type_id == 4) return 'Expense and Overhead Budget';
    if (this.bdgt_type_id == 5) return 'Sales Qty Budget';
    if (this.bdgt_type_id == 6) return 'Common Budget';
    if (this.bdgt_type_id == 7) return 'Capital Budget';

  }
  onToolbarPreparing(e) {
    e.toolbarOptions.items.unshift({
      location: 'after',
      widget: 'dxButton',
      options: {
        icon: 'refresh',
        onClick: this.refreshDataGrid.bind(this),
      },
    });
  }
  refreshDataGrid() {
    this.selectedCalender = this.commonSettingService.getSelectedCalender();
    if (this.authService.isPBCPM()) {
      this.getAllBudget(this.selectedCalender.calendarId, 0, 0, 1);
    } else {
      // this.bdgt_type_id = 0;
      this.getAllBudget(this.selectedCalender.calendarId, this.sessionUser.hr_employee_id, this.bdgt_type_id, this.actionPick);

    }

    //this.dataGrid.instance.refresh();
  }

  async getAllBudget(calId, userId, typeId, actionPick: number) {
    this.globalService.showSpinner(true);
    (await this.budgetService.getBudgetListBy(calId, userId, typeId)).subscribe(
      (result) => {
        if (result.statusCode == 200) {
          // this.programList = result.resultset;
          this.dataSource = result.result == null ? [] : result.result;
          if(this.dataSource?.length != 0 || this.dataSource){

            if (actionPick == 2) {
              // Only Pending by user
              this.dataSource = this.dataSource.filter(x => x.statusbyhrid == this.sessionUser.hr_employee_id)
            }
            // if (!this.isPBCPMRole) {
              //   // Only Pending by user
              //   this.dataSource = this.dataSource.filter(x => x.statusby == this.sessionUser.user_name)
              // }
              let slNo = 1;
              this.dataSource?.forEach((element) => {
                element.slNo = slNo++;
              });
              this.globalService.showSpinner(false);
            } else {
              this.globalService.showSpinner(false);
            }
        }
      },
      (err) => {
        this.globalService.showSpinner(false);
        console.log(err);
      }
    );
  }
  update(data) {
    const url = this.budgetService.getBudgetTypeNameById(data.budgetTypeId).routerUrl + '/create?id=' + data.budgetId;
    //this.router.navigateByUrl(url);
    //this.router.navigate(['./createCohort?id='+ id]);
    window.open(url, "_blank");
  }

  delete(id) {
    Swal.fire({
      title: 'Are you sure?',
      text: 'You want to delete!',
      icon: 'warning',
      showCancelButton: true,
      confirmButtonColor: '#3085d6',
      cancelButtonColor: '#d33',
      confirmButtonText: 'Yes, delete it!',
    }).then((result) => {
      /* Read more about isConfirmed, isDenied below */
      if (result.isConfirmed) {
        // this.masterDataManagementService.deleteCohort(id).subscribe(
        //   (result) => {
        //     console.log(result);
        //     if (result.status == 'ok') {
        //       if (result.result.is_success == false) {
        //         this.globalService.showSwal(
        //           'Not Deleted!!',
        //           result.result.message,
        //           'info'
        //         );
        //       } else {
        //         this.getCohortListByProgramID();
        //       }
        //     }
        //   },
        //   (err) => {
        //     console.log(err);
        //   }
        // );
      }
    });
  }

  actionListChange(e) {
    console.log(e);
    this.actionPick = e.selectedItem.id;
    this.refreshDataGrid();
  }
}

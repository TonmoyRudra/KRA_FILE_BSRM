import { Component, OnInit } from '@angular/core';
import { Router } from '@angular/router';
import { AuthService } from 'src/app/services/auth/auth.service';
import { BudgetService } from 'src/app/services/budget/budget.service';
import { CommonSettingService } from 'src/app/services/common-setting/common-setting.service';
import { GlobalService } from 'src/app/services/global/global.service';

@Component({
  selector: 'app-welcome',
  templateUrl: './welcome.component.html',
  styleUrls: ['./welcome.component.css']
})
export class WelcomeComponent implements OnInit {
  sessionUser: any;
  isPBCPMRole: boolean;
  allBudgetByUser: any = [];
  pendingBudgetByUser: any = [];
  selectedCalender: any;

  constructor(
    public authService:AuthService,
    public router: Router,
    public globalService: GlobalService,
    public commonSettingService: CommonSettingService,
    public budgetService:BudgetService) {
    this.sessionUser = this.authService.getSessionUser();
    this.selectedCalender = this.commonSettingService.getSelectedCalender();
    this.isPBCPM();
  }

  ngOnInit(): void {
    this.getAllBudget(this.selectedCalender.calendarId, this.sessionUser.hr_employee_id, 0, 1);
  }

  isPBCPM(){
    this.isPBCPMRole = this.authService.isPBCPM();
    return this.authService.isPBCPM();
  }

  async getAllBudget(calId, userId, typeId, actionPick: number) {
    this.globalService.showSpinner(true);
    (await this.budgetService.getBudgetListBy(calId, userId, typeId)).subscribe(
      (result) => {
        if (result.statusCode == 200) {
          // this.programList = result.resultset;
          if(result.result?.length != 0 || result.result){

            const allBudget = result.result == null ? [] : result.result;
            this.allBudgetByUser = allBudget;
            this.pendingBudgetByUser = allBudget.filter(x => x.statusbyhrid == this.sessionUser.hr_employee_id)

            this.globalService.showSpinner(false);
          } else{
            this.globalService.showSpinner(false);

          }
        }
      },
      (err) => {
        this.globalService.showSpinner(false);
        console.log(err);
      }
    );
  }
}

import { BudgetFlowMaster } from './../../models/budgetFlowMaster-dto.model';

import { NgxSpinnerService } from 'ngx-spinner';
import { AuthService } from 'src/app/services/auth/auth.service';
import { BudgetService } from 'src/app/services/budget/budget.service';
import { CommonSettingService } from 'src/app/services/common-setting/common-setting.service';
import { ExpenseService } from 'src/app/services/expense/expense.service';
import { GlobalService } from 'src/app/services/global/global.service';
import { PbcpmService } from 'src/app/services/pbcpm/pbcpm.service';
import { Subject } from 'rxjs';
import { takeUntil } from 'rxjs/operators';
import ArrayStore from "devextreme/data/array_store";
@Component({
  selector: 'app-workflow-action-log-history',
  templateUrl: './workflow-action-log-history.component.html',
  styleUrls: ['./workflow-action-log-history.component.css']
})
export class WorkflowActionLogHistoryComponent implements OnInit {
  @Input() budgetId;
  sessionUser: any;
  budgetLogHistory: any;
  budgetMasterInfo: any;
  userRoleForThisBudget: string | { roleName: string; statusId: string; isEditor: boolean; };
  constructor(private spinner: NgxSpinnerService,
    private authService: AuthService,
    public globalService: GlobalService,
    private pbcpmService: PbcpmService,
    public router: Router,
    public route: ActivatedRoute,
    private commonSettingService: CommonSettingService,
    private expenseService: ExpenseService,
    private budgetService: BudgetService) {
    this.sessionUser = this.authService.getSessionUser();


  }

  ngOnInit(): void {
    console.log("app-workflow-action-log-history")
    //this.getBudgetLogHistoryInfo(this.budgetId)
    this.refreshDataGrid();
  }

  onToolbarPreparing(e) {
    e.toolbarOptions.items.unshift({
      location: 'after',
      widget: 'dxButton',
      options: {
        icon: 'refresh',
        onClick: this.refreshDataGrid.bind(this),
      },
    });
  }

  refreshDataGrid() {
    this.getBudgetMasterInfo(this.budgetId);
    this.getBudgetLogHistoryInfo(this.budgetId);
    this.budgetService.setWorkFlowUILoadObs(true);
  }

  async getBudgetMasterInfo(bdgtId) {
    this.globalService.showSpinner(true);
    (await this.budgetService.getBudgetMasterInfoBy(bdgtId)).subscribe(
      (result) => {
        if (result.statusCode == 200) {
          // this.programList = result.resultset;
          this.budgetMasterInfo = result.result;
          console.log(this.budgetMasterInfo);
          this.userRoleForThisBudget = this.budgetService.getUserRoleForThisBudget(this.budgetMasterInfo);
          console.log('role',this.userRoleForThisBudget)
          this.globalService.showSpinner(false);
        }
      },
      (err) => {
        this.globalService.showSpinner(false);
        console.log(err);
      }
    );
  }
  async getBudgetLogHistoryInfo(bdgtId) {
    //this.globalService.showSpinner(true);
    (await this.budgetService.getBudgetLogHistoryBy(bdgtId)).subscribe(
      (result) => {
        if (result.statusCode == 200) {
          // this.programList = result.resultset;
          this.budgetLogHistory = result.result;
          console.log(this.budgetLogHistory);
          let slNo = 1;
          this.budgetLogHistory?.forEach((element) => {
            element.slNo = slNo++;
          });

        }
      },
      (err) => {
        //this.globalService.showSpinner(false);
        console.log(err);
      }
    );
  }
}

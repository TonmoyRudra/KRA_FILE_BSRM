    this.popUpTitle = "Workflow Action Log History";

        icon: 'refresh',

      },
    });

  async refreshDataGrid() {
    await this.getBudgetMasterInfo(this.budget_id);
    this.getStoreSpareBudgetById(this.budget_id);
     // This is for get production Budget list form all budget
    //await this.getAllBudget(this.selectedCalender.calendarId, this.sessionUser.hr_employee_id, 0, 1);

    this.budgetService.setWorkFlowUILoadObs(true);
    this.comments = null;
    //this.generateCompanyWiseTotal();

  }

  getCalenderDetailsByCalID(calId) {
     this.commonSettingService.getCalendarById(calId).subscribe(
      (result) => {
        if (result.statusCode == 200) {
          // this.programList = result.resultset;
          //this.generateGridsource(result.result);
          this.selectedBudgetCalDetails =  result.result;
          this.fiscalYearList = this.getFiscalYear(this.selectedBudgetCalDetails.fromDate);
        }
      },
      (err) => {
        this.globalService.showSpinner(false);
        console.log(err);
      }
    );
  }

  async getBudgetMasterInfo(bdgtId) {
    this.globalService.showSpinner(true);
    (await this.budgetService.getBudgetMasterInfoBy(bdgtId)).subscribe(
      (result) => {
        if (result.statusCode == 200) {

          // this.programList = result.resultset;
          this.budgetMasterInfo = result.result;
          console.log(this.budgetMasterInfo);
          this.getCalenderDetailsByCalID(this.budgetMasterInfo.calendarId);
          this.userRoleForThisBudget = this.budgetService.getUserRoleForThisBudget(this.budgetMasterInfo);
          console.log('role',this.userRoleForThisBudget)
          this.editingPermissionGenerate();
          this.globalService.showSpinner(false);
          // const element :any  =document.getElementsByClassName("dx-datagrid-rowsview")
          // element.style.height = "450px";

          // this.getOrgList(this.budgetMasterInfo.orgId)
          //get tagged file
          this.getStoreSpareFileList();
          return this.budgetMasterInfo;
        }
      },
      (err) => {
        this.globalService.showSpinner(false);
        console.log(err);
        return null;
      }
    );
  }

  editingPermissionGenerate(){
    this.allowUpdating = false;
    this.editCellForPBCPM = false;
    this.editCellForPreparer = false;

    if(this.userRoleForThisBudget.roleName == 'preparer' && this.userRoleForThisBudget.statusId == 0){
      this.allowUpdating = true;
      this.editCellForPreparer = true;
    }
    if(this.userRoleForThisBudget.roleName == 'pbcpm' && this.userRoleForThisBudget.statusId == 11){
      this.allowUpdating = true;
      this.editCellForPBCPM = true;
    }
  }

    if(dxSaveButton){
      dxSaveButton.style.display = 'none';
    }
  }

  getFiscalYear(currentCalFromDate: string) {
    var fiscalyearFrom;
    var fiscalyearTo;
    var fiscalyear;
    var last1;
    var last2;
    var last3;
    var today = new Date(currentCalFromDate);

    if ((today.getMonth() + 1) <= 6) {
      fiscalyearFrom = (today.getFullYear() - 1);
      fiscalyearTo = (today.getFullYear());
      fiscalyear = fiscalyearFrom + "-" + fiscalyearTo;

  onFileChange(event: any) {
    /* wire up file reader */

    const target: DataTransfer = <DataTransfer>event.target;
    // if (target.files.length !== 1) {
    //   this.globalService.showSwal('Warning!!' ,'Cannot use multiple files', 'warning');
    //   throw new Error('Cannot use multiple files');
    // }

    if (target.files[0]) {
      this.fileData = target.files[0];
      if (this.fileData.name.split('.').pop() != 'xlsx') {
        this.globalService.showSwal(
          'Warning!!',
          'Please choose only .xlsx file format.',
          'warning'
        );
      } else {
        this.uploadedDataSource = [];
        const reader: FileReader = new FileReader();
        reader.readAsBinaryString(this.fileData);
        reader.onload = (e: any) => {
          /* create workbook */
          const binarystr: string = e.target.result;
          const wb: XLSX.WorkBook = XLSX.read(binarystr, { type: 'binary' });

          /* selected the first sheet */
          const wsname: string = wb.SheetNames[0];
          const ws: XLSX.WorkSheet = wb.Sheets[wsname];

          /* save data */
          const data = XLSX.utils.sheet_to_json(ws); // to get 2d array pass 2nd parameter as object {header: 1}
          let slNo = 1;
          this.uploadedDataSource = data;
          // this.datasource.forEach((element) => {
          //   element.slNo = slNo++;
          // });

          this.columns = []; // Make Column for Grid
          if (this.uploadedDataSource.length != 0) {
            var key = Object.keys(this.uploadedDataSource[0]);
            key.forEach((element) => {
              if (element == 'slNo') {
                this.columns.unshift(element);
              } else {
                this.columns.push(element);
              }
            });
          }
          this.uploadedDataSource.splice(0,1); // slice 1st row [0 index], because of dummy info text
          console.log(this.uploadedDataSource); // Data will be logged in array format containing objects

          this.datagrid_DXSaveButton_Disable();
        };
      }
    }
    // Clear the file picker.
    let fileDropElemRef: any = document.getElementById("fileDropRef");
    fileDropElemRef.value = "";
  }

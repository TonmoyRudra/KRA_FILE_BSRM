@Component({
  selector: 'app-calender-add',

export class CalenderAddComponent implements OnInit {
  calendarId: any;
  editMode: boolean;
  model: any = {
    calendarId : null,
    calendarName : null,
    fromDate : null,
    toDate : null,

    remarks: null,

    createdBy: null,
    creationDate:null,
    lastUpdateBy: null,
    lastUpdateDate: null,
  };


  constructor(  private route: ActivatedRoute, public commonSettingService: CommonSettingService,
    public globalService: GlobalService, private datePipe: DatePipe) {

    }

  ngOnInit(): void {
    this.route.queryParams.subscribe((params) => {
      this.calendarId = params['calendarId'];
      if (this.calendarId) {
        this.editMode = true;
        this.getCalendarById(this.calendarId);
      }
    });
  }

  getCalendarById(id) {
    this.commonSettingService.getAllCalender(id).subscribe(
      (result) => {
        if (result.statusCode == 200) {
          console.log(result.result);
          // var s = result.result.map((iCalendar : ICalendar) => {
          //   return iCalendar;
          // });

          this.model = result.result[0];
          console.log(this.model);
        }
      },
      (err) => {
        console.log(err);
      }
    );
  }

  activeModeChange(data, e) {
    console.log(data);
    this.model.inactive = data == 'Y' ? 'N' : 'Y';
    // let calenderID = data.calendarId;

    //     // }
    //     this.getCohortListByProgramID();
    //   },
    //   (err) => {
    //     console.log(err);
    //   }
    // );
  }
  onSubmit(e){
    const { isValid } = ValidationEngine.validateGroup('validationGrp');
    if (isValid) {
      console.log(this.model);
      if(this.editMode){
        // PUT
        this.commonSettingService.updateCalender(this.model).subscribe(
          (result) => {
            if(result.statusCode == 200){
              console.log(result.result);
              this.globalService.showSwal('SUCCESS', 'Successfully update a calender', 'success');
              this.clearField();
            }
          },
          (err) => {
            console.log(err);
          });
      } else {
        // POST
      delete this.model.calendarId;

      this.model.fromDate = this.model.fromDate.toLocaleString();
      this.model.toDate = this.model.toDate.toLocaleString();
      this.model.openDate = this.model.openDate.toLocaleString();
      this.model.closeDate = this.model.closeDate.toLocaleString();

      this.commonSettingService.createCalender(this.model).subscribe(
      (result) => {
        if(result.statusCode == 200){
          console.log(result.result);
          this.globalService.showSwal('SUCCESS', 'Successfully create a calender', 'success');
          this.clearField();
        }
      },
      (err) => {
        console.log(err);
        this.globalService.showSwal('ERROR', err, 'error');

      });

      }
    }
  }
  clearField() {
    ValidationEngine.resetGroup('validationGrp');
    //this.model = {};
  }

}

import { StoreSpareBdgtDto } from './../../shared/models/storeSpare-dto.model';
import { RMPMBdgtDto } from '../../shared/models/rmpm-dto.model';
import { HttpClient } from '@angular/common/http';
import { Injectable } from '@angular/core';
import { Router } from '@angular/router';
import { Observable, throwError } from 'rxjs';
import { catchError, map } from 'rxjs/operators';
import { GlobalService } from '../global/global.service';

@Injectable({
  providedIn: 'root'
})
export class StoreSpareBdgtService {

  apiURL = this.globalService.api_url + '/StoreSpareBudget';

	constructor(
    private http: HttpClient,
    public router : Router,
    public globalService: GlobalService) { }

    getStoreSpareBdgtDetailsById(bdgtId) {
      return this.http.get(this.apiURL + '/getDetailsById?id=' + bdgtId).pipe(
        map((x: Response) => x),
        map((x: any) => x),
        catchError((error: Response) => {
          return throwError(error);
        })
      );
    }

    updateStoreSpareBudget(model : StoreSpareBdgtDto) {
      return this.http.post(this.apiURL + '/update', model).pipe(
        map((x: Response) => x),
        map((x: any) => x),
        catchError((error: Response) => {
          return throwError(error);
        })
      );
    }

    uploadStoreSpareBudget(model:any) {
      return this.http.post(this.apiURL + '/uploadFreshdata', model).pipe(
        map((x: Response) => x),
        map((x: any) => x),
        catchError((error: Response) => {
          return throwError(error);
        })
      );
    }


    getAllStoreFile(id = 0, locCode = "", costCode = "") {
      return this.http.get(this.apiURL + '/storeBdgtFile-list?pid=' + id + '&locCode=' +locCode + '&costCode=' + costCode ).pipe(
        map((x: Response) => x),
        map((x: any) => x),
        catchError((error: Response) => {
          return throwError(error);
        })
      );
    }

    createStoreFile(body) {
      return this.http.post(this.apiURL + '/storeBdgtFile/', body).pipe(
        map((x: Response) => x),
        map((x: any) => x),
        catchError((error: Response) => {
          return throwError(error);
        })
      );
    }

    editStoreFile(body) {
      return this.http.put(this.apiURL + '/storeBdgtFile/', body).pipe(
        map((x: Response) => x),
        map((x: any) => x),
        catchError((error: Response) => {
          return throwError(error);
        })
      );
    }
    deleteStoreFile(id): Observable<any> {
      return this.http.delete(this.apiURL + '/storeBdgtFile?pid='+id).pipe(
        map((x: Response) => x),
        map((x: any) => x),
        catchError((error: Response) => {
          return throwError(error);
        })
      );
    }
}

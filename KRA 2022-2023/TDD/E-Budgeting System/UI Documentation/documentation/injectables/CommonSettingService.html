import { HttpClient } from '@angular/common/http';
import { Injectable } from '@angular/core';
import { Router } from '@angular/router';
import { Observable, throwError } from 'rxjs';
import { catchError, map } from 'rxjs/operators';
import { GlobalService } from '../global/global.service';

@Injectable({
  providedIn: 'root'
})
export class CommonSettingService {

  commonSettingServiceURL = this.globalService.api_url + '/Common';

	constructor(
    private http: HttpClient,
    public router : Router,
    public globalService: GlobalService) { }

    getAllCalender(calId) {
      return this.http.get(this.commonSettingServiceURL + '/calendar-list?calId=' + calId).pipe(
        map((x: Response) => x),
        map((x: any) => x),
        catchError((error: Response) => {
          return throwError(error);
        })
      );
    }

    getCalendarById(calId) {
      return this.http.get(this.commonSettingServiceURL + '/calendar-details?calId=' + calId).pipe(
        map((x: Response) => x),
        map((x: any) => x),
        catchError((error: Response) => {
          return throwError(error);
        })
      );
    }

    createCalender(model) {
      return this.http.post(this.commonSettingServiceURL + '/calender', model).pipe(
        map((x: Response) => x),
        map((x: any) => x),
        catchError((error: Response) => {
          return throwError(error);
        })
      );
    }

    updateCalender(model) {
      return this.http.put(this.commonSettingServiceURL + '/calender', model).pipe(
        map((x: Response) => x),
        map((x: any) => x),
        catchError((error: Response) => {
          return throwError(error);
        })
      );
    }


    deleteCalender(calId) {
      return this.http.delete(this.commonSettingServiceURL + '/calender?calId=' + calId).pipe(
        map((x: Response) => x),
        map((x: any) => x),
        catchError((error: Response) => {
          return throwError(error);
        })
      );

    }

    setSelectedCalender(data){
      localStorage.setItem('selectedCalender', JSON.stringify(data));
    }

    getSelectedCalender(){
      return  JSON.parse(localStorage.getItem('selectedCalender'));
    }

    selectedCalenderBudgetPeriodOpen(): boolean {
      let selectedCalender = this.getSelectedCalender();
      // let dateOptions = {dateStyle: 'short'} as const;
      const dateOptions: any = {dateStyle: "short"};

      //let today = new Date().toLocaleString('en-IN', dateOptions);
      let today = new Date();
      console.log('today',today);
      //let selectedCalerderCloseDate =  new Date(selectedCalender.closeDate).toLocaleString('en-IN', dateOptions);
      let selectedCalerderCloseDate =  new Date(selectedCalender.closeDate);
      console.log('clsoedate',selectedCalerderCloseDate);
      selectedCalerderCloseDate = new Date(selectedCalerderCloseDate.setTime(selectedCalerderCloseDate.getTime() + (86340 * 1000)));
      //console.log('newcloseDate',new Date(selectedCalerderCloseDate.setTime(selectedCalerderCloseDate.getTime() + (3600 * 1000))));

      if(selectedCalerderCloseDate >= today) {
        console.log('Budget period open ');
        return true;
      } else {
        console.log('Budget period Close ');
        return false;

      }
    }

    getHrEmployees() {
      return this.http.get(this.commonSettingServiceURL + '/hr-employee-list').pipe(
        map((x: Response) => x),
        map((x: any) => x),
        catchError((error: Response) => {
          return throwError(error);
        })
      );
    }

    getAssetList() {

          return throwError(error);
        })
      );
    }

        map((x: any) => x),

          return throwError(error);
        })
      );
    }

        map((x: Response) => x),
        map((x: any) => x),
        catchError((error: Response) => {
          return throwError(error);
        })
      );
    }


    deleteAsset(id) {
      return this.http.delete(this.commonSettingServiceURL + '/asset?id=' + id).pipe(
        map((x: Response) => x),
        map((x: any) => x),
        catchError((error: Response) => {
          return throwError(error);
        })
      );
    }

    getAssetCategoryList() {
      return this.http.get(this.commonSettingServiceURL + '/asset-category-list').pipe(

}

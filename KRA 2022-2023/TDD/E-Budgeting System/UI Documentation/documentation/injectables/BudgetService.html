import { CommonSettingService } from 'src/app/services/common-setting/common-setting.service';
import { WorkFlow } from './../../shared/models/workflow-dto.model';
import { AuthService } from './../auth/auth.service';
import { HttpClient } from '@angular/common/http';
import { Injectable } from '@angular/core';
import { Router } from '@angular/router';
import { BehaviorSubject, Observable, throwError } from 'rxjs';
import { catchError, map } from 'rxjs/operators';
import { GlobalService } from '../global/global.service';
@Injectable({
  providedIn: 'root'
})
export class BudgetService {
  private workFlowUILoadObs$: BehaviorSubject<boolean> = new BehaviorSubject(true);

  budgeterviceURL = this.globalService.api_url + '/Budget';
  sessionUser: any;
  budgetMasterInfo: any;

	constructor(
    private http: HttpClient,
    public router : Router,
    public globalService: GlobalService,
    public authService: AuthService, public commonSettingService :  CommonSettingService) { }

    getWorkFlowUILoadObs(): Observable<any> {
      return this.workFlowUILoadObs$.asObservable();
    }
    setWorkFlowUILoadObs(data : boolean) {
      this.workFlowUILoadObs$.next(data);
    }

    getBudgetListBy(calId, userid,typeId) {
      return this.http.get(this.budgeterviceURL + '/budget-list?calenderId=' + calId + '&userId=' + userid + '&typeId=' + typeId).pipe(
        map((x: Response) => x),
        map((x: any) => x),
        catchError((error: Response) => {
          return throwError(error);
        })
      );
    }

    getBudgetMasterInfoBy(bdgtId) :Observable<any> {
      return this.http.get(this.budgeterviceURL + '/budget-master-info?bdgtId=' + bdgtId).pipe(
        map((x: Response) => x),
        map((x: any) => {
          this.budgetMasterInfo = x.result;
          return x;
        }),
        catchError((error: Response) => {
          return throwError(error);
        })
      );
    }

    getBudgetWorkflowInfoBy(bdgtId) {
      return this.http.get(this.budgeterviceURL + '/budget-workflow-info?bdgtId=' + bdgtId).pipe(
        map((x: Response) => x),
        map((x: any) => x),
        catchError((error: Response) => {
          return throwError(error);
        })
      );
    }

    getBudgetLogHistoryBy(bdgtId) {
      return this.http.get(this.budgeterviceURL + '/budget-log-history?bdgtId=' + bdgtId).pipe(
        map((x: Response) => x),
        map((x: any) => x),
        catchError((error: Response) => {
          return throwError(error);
        })
      );
    }

    getUserRoleForThisBudget(budgetMasterInfo){
      var statusId = budgetMasterInfo.statusid;
      this.sessionUser = this.authService.getSessionUser();
      let obj = {
        roleName: '',
        statusId:'',
        isEditor: false

      // }

        obj.roleName = 'preparer';

        return  obj;

    updateWorkflow(model:WorkFlow) {
      return this.http.post(this.budgeterviceURL + '/budget-workflow-update', model).pipe(
        map((x: Response) => x),
        map((x: any) => x),
        catchError((error: Response) => {
          return throwError(error);
        })
      );
    }

    workflowStepChange(bdgtId: number, step: string, toStepId: number) {
      return this.http.post(this.budgeterviceURL + '/budget-workflow-step-change?bdgtId='+bdgtId +'&step=' + step + '&toStepid=' + toStepId, null).pipe(
        map((x: Response) => x),
        map((x: any) => x),
        catchError((error: Response) => {
          return throwError(error);
        })
      );
    }

    getBudgetTypeNameById(typeId){

      if(typeId == 1){
        return {typeId: typeId, typeName: 'Production Budget', routerUrl: 'main/productionBudget'}
      } else if(typeId == 2){
        return {typeId: typeId, typeName: 'Raw Packing Consumable Consumption Budget', routerUrl: 'main/rmpmBudget'}
      } else if(typeId == 3){
        return {typeId: typeId, typeName: 'Store and Spare Budget', routerUrl: 'main/storespareBudget'}
      } else if(typeId == 4){
        return {typeId: typeId, typeName: 'Expense and Overhead Budget', routerUrl: 'main/expenseBudget'}
      } else if(typeId == 5){
        return {typeId: typeId, typeName: 'Sales Qty Budget', routerUrl: 'main/salesBudget'}
      } else if(typeId == 6){
        return {typeId: typeId, typeName: 'Common Budget', routerUrl: 'main/commonBudget'}
      } else if(typeId == 7){
        return {typeId: typeId, typeName: 'Capital Budget', routerUrl: 'main/capitalBudget'}
      } else {
        return {typeId: typeId, typeName: 'Not Found Any Budget type name', routerUrl: ''}
      }
    }
}
